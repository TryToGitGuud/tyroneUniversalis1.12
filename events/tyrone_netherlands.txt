	}

		c:NET = { save_scope_as = netherlands_scope }

		c:DEI = { save_scope_as = dei_scope }

	}

	duration = 1

		name = dei_events.3.a
		default_option = yes
		c:DEI = {
			add_modifier = {
				name = DEI_colonial_rule_supported
				days = long_modifier_time
			}
			add_loyalists = {
				value = very_large_radicals
				culture = cu:dutch
			}
		}
	}
	option = { # Reform the VOC
		name = dei_events.3.b
		default_option = yes
		c:DEI = {
			if = {
				limit = {
					NOT = { is_subject_type = subject_type_chartered_company }
				}
				change_subject_type = subject_type_chartered_company
			}
			set_variable = VOC_reformed
		}
		add_modifier = {
			name = VOC_reformed_modifier
			days = very_long_modifier_time
		}
		company:company_nhm = {
			add_owned_country = c:DEI #links the country to this company
		}
	}
	option = { # Push for liberal reform
		name = dei_events.3.c
		default_option = yes
		c:DEI = {
			activate_law = law_type:law_slavery_banned
			add_modifier = {
				name = DEI_liberal_reforms_modifier
				days = long_modifier_time
			}
			add_loyalists = {
				value = medium_radicals
			}
		}
	}
}

# The New Feudalism
dei_events.6 = {
	type = country_event
	placement = ROOT

	title = dei_events.6.t
	desc = dei_events.6.d
	flavor = dei_events.6.f

	event_image = {
		video = "asia_sepoy_mutiny"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	duration = 3

	trigger = {
	}

	immediate = {
	}

	option = {
		name = dei_events.6.a
		default_option = yes
		add_journal_entry = {
			type = je_ethical_policy_feudalism
		}
	}

	after = {
		if = {
			limit = {
				has_law_or_variant = law_type:law_agrarianism
			}
			activate_law = law_type:law_neofeudalism
		}
		else = {
			active_law:lawgroup_economic_system = {
				add_amendment = {
					type = amendment_colonial_neofeudalism
					sponsor = PREV.ig:ig_landowners
				}
			}
		}
		if = {
			limit = {
				has_law = law_type:law_slave_trade
			}
			active_law:lawgroup_slavery = {
				add_amendment = {
					type = amendment_expanded_colonial_slavery
					sponsor = PREV.ig:ig_landowners
				}
			}
		}
	}
}
