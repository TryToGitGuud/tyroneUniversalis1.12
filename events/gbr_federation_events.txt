# Milestone events fired by je_imperial_federation's on_monthly_pulse when bar crosses thresholds.

# Threshold 10 — Colonial Conference
gbr_fed_events.1 = {
	type = country_event
	duration = 3

	title = gbr_fed_events.1.t
	desc = gbr_fed_events.1.d
	flavor = gbr_fed_events.1.f

	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	option = {
		name = gbr_fed_events.1.a
		default_option = yes
		every_subject_or_below = {
			change_relations = { country = c:GBR value = 20 }
		}
	}

	option = {
		name = gbr_fed_events.1.b
		every_subject_or_below = {
			add_liberty_desire = -10
		}
	}
}

# Threshold 90 — Australia Speaks
gbr_fed_events.6 = {
	type = country_event
	duration = 3

	title = gbr_fed_events.6.t
	desc = gbr_fed_events.6.d
	flavor = gbr_fed_events.6.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	immediate = {
		if = {
			limit = { exists = c:AST }
			c:AST = { save_scope_as = aus_dominion_scope }
		}
		else_if = {
			limit = { exists = c:NSW }
			c:NSW = { save_scope_as = aus_dominion_scope }
		}
	}

	option = {
		name = gbr_fed_events.6.a
		default_option = yes
		if = {
			limit = { exists = scope:aus_dominion_scope }
			scope:aus_dominion_scope = {
				change_relations_value = { country = c:GBR value = 50 }
				change_liberty_desire = -50
			}
		}
		ruler = {
			add_modifier = {
				name = imperial_federation_support_from_australia
				years = 5
			}
		}
	}
}

# Threshold 45 — The Point of No Return
gbr_fed_events.7 = {
	type = country_event
	duration = 3

	title = gbr_fed_events.7.t
	desc = gbr_fed_events.7.d
	flavor = gbr_fed_events.7.f

	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	option = {
		name = gbr_fed_events.7.a
		trigger = {
			NOT = { has_law = law_type:law_cultural_exclusion }
			NOT = { has_law = law_type:law_multicultural }
			NOT = { is_enacting_law = law_type:law_multicultural }
		}
		activate_law = law_type:law_cultural_exclusion
	}

	option = {
		name = gbr_fed_events.7.b
		trigger = {
			NOT = { has_law = law_type:law_no_migration_controls }
		}
		activate_law = law_type:law_no_migration_controls
	}

	option = {
		name = gbr_fed_events.7.c
		trigger = {
			OR = {
				has_law = law_type:law_cultural_exclusion
				has_law = law_type:law_multicultural
				is_enacting_law = law_type:law_multicultural
			}
			has_law = law_type:law_no_migration_controls
		}
		every_scope_state = {
			add_loyalists = {
				value = 0.05
				strata = middle
			}
			add_loyalists = {
				value = 0.05
				strata = rich
			}
		}
	}
}
