namespace = ostend_manifesto
### Ostend Manifesto Events
# Spain receives the American offer to purchase Cuba
ostend_manifesto.1 = {
	type = country_event
	placement = root

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = ostend_manifesto.1.t
	desc = ostend_manifesto.1.d
	flavor = ostend_manifesto.1.f

	duration = 3

	trigger = {
		exists = c:USA
		exists = c:CUB
	}

	option = {
		name = ostend_manifesto.1.a
		# Accept: receive 5M over 5 years
		add_modifier = {
			name = ostend_cuba_offer_income
			months = 60
		}
		c:USA = {
			add_modifier = {
				name = ostend_cuba_offer_payment
				months = 60
			}
			trigger_event = { id = ostend_manifesto.3 days = 7 }
		}
		ai_chance = {
			base = 0
			modifier = {
				trigger = { "relations(c:USA)" > relations_threshold:cordial }
				add = 100
			}
		}
	}

	option = {
		name = ostend_manifesto.1.b
		# Counteroffer: 7.5M
		c:USA = {
			trigger_event = { id = ostend_manifesto.2 days = 7 }
		}
		ai_chance = {
			base = 0
		}
	}

	option = {
		name = ostend_manifesto.1.c
		# Refuse
		change_relations = { country = c:USA value = -25 }
		c:USA = { set_variable = ostend_cuba_offer_refused }
		ai_chance = {
			base = 0
			modifier = {
				trigger = { "relations(c:USA)" <= relations_threshold:cordial }
				add = 100
			}
		}
	}
}

# Treaty of Havana — USA decides Cuba's political status
ostend_manifesto.3 = {
	type = country_event
	placement = s:STATE_WESTERN_CUBA.region_state:CUB

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = ostend_manifesto.3.t
	desc = ostend_manifesto.3.d
	flavor = ostend_manifesto.3.f

	duration = 3

	option = {
		name = ostend_manifesto.3.a
		# Grant Cuba full sovereignty
		c:SPA = {
			remove_diplomatic_pact = { country = c:CUB type = colony }
		}
		change_relations = { country = c:CUB value = 50 }
	}

	option = {
		name = ostend_manifesto.3.b
		# Establish American protectorate
		c:SPA = {
			remove_diplomatic_pact = { country = c:CUB type = colony }
		}
		create_diplomatic_pact = { country = c:CUB type = protectorate }
	}

	option = {
		name = ostend_manifesto.3.c
		# Annex Cuba as a colony
		c:SPA = {
			remove_diplomatic_pact = { country = c:CUB type = colony }
		}
		create_diplomatic_pact = { country = c:CUB type = colony }
		create_incident = { country = c:CUB value = 10 }
	}
}
