namespace = intl_communism

# Replaces the vanilla communism.2 ideology choice with the five Internationale currents.
intl_communism.2 = {
	type = country_event
	placement = root
	title = intl_communism.2.t
	desc = intl_communism.2.d
	flavor = intl_communism.2.f

	duration = 3

	event_image = {
		video = "votp_agitator_speaking"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	trigger = {
		# Triggered by je_communism_2, or directly by je_communism_1 when Council Republic already exists.
	}

	immediate = {
		ig:ig_trade_unions ?= {
			save_scope_as = unions
			leader = {
				save_scope_as = union_leader
			}
		}
	}

	option = {
		name = communism.2.anarchist
		set_variable = bakunin_var
		intl_unlock_anarchist_laws = yes
		intl_set_anarchist_movement_line = yes
		ig:ig_trade_unions ?= {
			set_variable = chose_anarchism
		}
		custom_tooltip = chose_internationale_anarchism.tt
		if = {
			limit = {
				scope:union_leader = {
					NOT = { has_ideology = ideology:ideology_anarchist }
				}
			}
			scope:union_leader = {
				set_ideology = ideology:ideology_anarchist
			}
		}
		add_modifier = {
			name = path_to_anarchy
			days = normal_modifier_time
		}
		if = {
			limit = {
				has_law_or_variant = law_type:law_anarchy
				average_sol >= 15
			}
			trigger_event = { id = communism.4 }
		}
		else = {
			add_journal_entry = { type = je_anarchy }
		}
	}

	option = {
		name = communism.2.mutualist
		set_variable = proudhon_var
		intl_unlock_mutualist_laws = yes
		intl_set_mutualist_movement_line = yes
		ig:ig_trade_unions ?= {
			set_variable = chose_communism
		}
		custom_tooltip = chose_internationale_mutualism.tt
		if = {
			limit = {
				scope:union_leader = {
					NOT = { has_ideology = ideology:ideology_mutualist }
				}
			}
			scope:union_leader = {
				set_ideology = ideology:ideology_mutualist
			}
		}
		add_modifier = {
			name = path_to_communism
			days = normal_modifier_time
		}
		if = {
			limit = {
				NOT = { has_journal_entry = je_mutualist_worker_councils }
				NOT = { has_variable = mutualist_worker_councils_founded }
			}
			add_journal_entry = { type = je_mutualist_worker_councils }
		}
		add_journal_entry = { type = je_communist_welfare }
	}

	option = {
		name = communism.2.syndicalist
		set_variable = bel_de_paepe_var
		intl_unlock_syndicalist_laws = yes
		intl_set_syndicalist_movement_line = yes
		ig:ig_trade_unions ?= {
			set_variable = chose_communism
		}
		custom_tooltip = chose_internationale_syndicalism.tt
		if = {
			limit = {
				scope:union_leader = {
					NOT = { has_ideology = ideology:ideology_syndicalist }
				}
			}
			scope:union_leader = {
				set_ideology = ideology:ideology_syndicalist
			}
		}
		add_modifier = {
			name = path_to_communism
			days = normal_modifier_time
		}
		add_journal_entry = { type = je_syndicalist_amendments }
	}

	option = {
		name = communism.2.blanquist
		set_variable = blanqui_var
		intl_unlock_blanquist_laws = yes
		intl_set_blanquist_movement_line = yes
		ig:ig_trade_unions ?= {
			set_variable = chose_vanguardism
		}
		custom_tooltip = chose_internationale_blanquism.tt
		if = {
			limit = {
				scope:union_leader = {
					NOT = { has_ideology = ideology:ideology_blanquist }
				}
			}
			scope:union_leader = {
				set_ideology = ideology:ideology_blanquist
			}
		}
		add_modifier = {
			name = path_to_vanguardism
			days = normal_modifier_time
		}
		if = {
			limit = {
				NOT = { has_journal_entry = je_blanquism }
				NOT = { has_variable = blanquist_preparation_done }
			}
			add_journal_entry = { type = je_blanquism }
		}
	}

	option = {
		name = communism.2.marxist
		default_option = yes
		set_variable = karl_marx_var
		intl_unlock_marxist_laws = yes
		intl_set_marxist_movement_line = yes
		ig:ig_trade_unions ?= {
			set_variable = chose_communism
		}
		custom_tooltip = chose_internationale_marxism.tt
		if = {
			limit = {
				scope:union_leader = {
					NOT = { has_ideology = ideology:ideology_marxist }
				}
			}
			scope:union_leader = {
				set_ideology = ideology:ideology_marxist
			}
		}
		add_modifier = {
			name = path_to_communism
			days = normal_modifier_time
		}
		if = {
			limit = {
				NOT = { has_journal_entry = je_marxist_proletarian_state }
				NOT = { has_variable = marxist_proletarian_state_done }
			}
			add_journal_entry = { type = je_marxist_proletarian_state }
		}
		add_journal_entry = { type = je_communist_welfare }
	}
}

intl_communism.3 = {
	type = country_event
	placement = root
	title = intl_communism.3.t
	desc = intl_communism.3.d
	flavor = intl_communism.3.f

	duration = 3

	event_image = {
		video = "votp_agitator_speaking"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	trigger = {
	}

	immediate = {
		ig:ig_armed_forces ?= {
			leader = {
				set_ideology = ideology:ideology_blanquist
			}
		}
		ig:ig_intelligentsia ?= {
			leader = {
				save_scope_as = blanquist_revolutionary_director
				set_ideology = ideology:ideology_blanquist
			}
		}
		ig:ig_petty_bourgeoisie ?= {
			leader = {
				set_ideology = ideology:ideology_blanquist
			}
		}
		ig:ig_trade_unions ?= {
			leader = {
				set_ideology = ideology:ideology_blanquist
			}
		}
	}

	option = {
		name = intl_communism.3.a
		default_option = yes
		intl_unlock_blanquist_laws = yes
		set_variable = blanqui_var
		activate_law = law_type:law_blanquist_presidential_republic
		activate_law = law_type:law_blanquist_technocracy
		set_variable = blanquist_coup_executed
		set_variable = blanquist_coup_event_resolved
		scope:blanquist_revolutionary_director = {
			set_character_as_ruler = yes
			add_modifier = {
				name = blanquist_revolutionary_director_popularity
			}
		}
		if = {
			limit = {
				NOT = { has_journal_entry = je_blanquist_consolidation }
				NOT = { has_variable = blanquist_consolidation_done }
			}
			add_journal_entry = { type = je_blanquist_consolidation }
		}
		add_modifier = {
			name = modifier_blanquist_revolutionary_order
			years = 10
		}
		add_radicals = {
			value = 0.15
		}
	}
}

		ig:ig_petty_bourgeoisie ?= {

				save_scope_as = mutualist_pb_leader
			}
		}
	}

	option = {
		name = intl_communism.5.a
		default_option = yes
		add_radicals = {
			strata = middle
			value = 0.10
		}
		every_interest_group = {
			limit = {
				NOT = { is_interest_group_type = ig_petty_bourgeoisie }
				law_stance = {
					law = law_type:law_monarchy
					value > neutral
				}
			}
			add_modifier = {
				name = modifier_mutualist_monarchist_ig_backlash
				years = 5
			}
		}
	}

	option = {
		name = intl_communism.5.b
		add_radicals = {
			value = 0.05
		}
		ig:ig_petty_bourgeoisie ?= {
			add_modifier = {
				name = modifier_mutualist_petite_bourgeoisie_split
				years = 5
			}
		}
		scope:mutualist_pb_leader = {
			replace_character_roles = { add = { agitator } remove_all = yes }
		}
		create_character = {
			ig_leader = yes
			interest_group = ig:ig_petty_bourgeoisie
			ideology = ideology_fascist
			traits = {
				ambitious
				bigoted
			}
		}
	}
}

intl_communism.6 = {
	type = country_event
	placement = root

	title = intl_communism.6.t
	desc = intl_communism.6.d
	flavor = intl_communism.6.f

	category = enactment

	duration = 3

	event_image = {
		video = "votp_agitator_speaking"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	cooldown = { months = 12 }

	trigger = {
		OR = {
			is_enacting_law = law_type:law_cooperative_ownership
			is_enacting_law = law_type:law_mutualist_economy
		}
		
		enactment_phase < 2
	}

	immediate = {
		currently_enacting_law = {
			save_scope_as = current_law_scope
		}
	}

	cancellation_trigger = {
		NOT = { currently_enacting_law = scope:current_law_scope }
	}

	option = {
		name = intl_communism.6.a
		default_option = yes
		highlighted_option = yes
		add_enactment_phase = 1
	}
}
