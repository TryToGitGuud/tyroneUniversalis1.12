		}

			limit = {
				NOT = {
					OR = {
						has_variable = je_spa_economic_regeneration_master_completed_var

				}

				add_amendment = {
					type = amendment_economic_regeneration_1

				}
			}
		}
		set_variable = laissez_faire_expanded_var
	}

	option = {
		name = economic_reform.1.b

			has_law_or_variant = law_type:law_interventionism
