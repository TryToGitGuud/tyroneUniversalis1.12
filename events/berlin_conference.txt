berlin_conference.1 = {
	type = country_event

	title = berlin_conference.1.t
	desc = berlin_conference.1.d
	flavor = berlin_conference.1.f

	cooldown = { days = never_fire_again_modifier_time }

	event_image = {
		video = "ep1_power_blocs"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		if = {
			limit = {
				NOT = {
					is_involved_in_journal_entry = je_berlin_conference
				}
				is_in_geographic_region = geographic_region_europe
				OR = {
					has_technology_researched = civilizing_mission
					country_rank = rank_value:great_power
					AND = {
						OR = {
							has_law = law_type:law_colonial_resettlement
							has_law = law_type:law_colonial_exploitation
						}
						country_rank = rank_value:major_power
					}
				}
			}
			je:je_berlin_conference = {
				add_involved_country = ROOT
			}
		}
	}

	option = {
		name = berlin_conference.1.a
		if = {
 			limit = {
 				NOT = { has_technology_researched = civilizing_mission }
 				can_research = civilizing_mission
 			}
 			add_technology_progress = { progress = 5000 technology = civilizing_mission }
 		}
	}


}
