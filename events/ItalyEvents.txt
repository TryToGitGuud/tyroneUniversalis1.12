italy_tyrone_characters.1 = {
	type = country_event
	placement = root

	title = italy_tyrone_characters.1.t
	desc = italy_tyrone_characters.1.d
	flavor = italy_tyrone_characters.1.f

	gui_window = event_window_1char_tabloid
	left_icon = scope:mazzini_scope

	icon = "gfx/interface/icons/event_icons/italyFlag.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/political_extremism"

	duration = 3

	trigger = {
	}

	immediate = {	
		create_character = {
			template = agitator_giuseppe_mazzini
			on_created = {
				set_variable = italy_mazzini
				set_character_immortal = yes
			}
		}
		random_scope_character = {
			limit = {
				has_variable = italy_mazzini
			}
			save_scope_as = mazzini_scope
		}
		set_variable = {
			name = italian_sentiment_var
			value = 0
		}
		add_journal_entry = {
			type = italy_risorgimento_stage
		}
		je:italy_risorgimento_stage = {
			add_to_variable_list = {
				name = com_risorgimento_mazzini
				target = scope:mazzini_scope
			}
		}
		clear_global_variable_list = italy_capital_countries_list
		every_country = {
			limit = {
				capital = {
					is_in_geographic_region = geographic_region_italy
				}
			}
			add_to_global_variable_list = {
				name = italy_capital_countries_list
				target = THIS
			}
		}

		every_state = {
			limit = {
				OR = {
					state_region = s:STATE_SICILY
					state_region = s:STATE_CALABRIA
					state_region = s:STATE_CAMPANIA
					state_region = s:STATE_APULIA
					state_region = s:STATE_ABRUZZO
					}
				}
				state_region = { add_state_trait = state_trait_sluggish_south }
			}
		every_state = {
			limit = {
				state_region = s:STATE_LAZIO
				}
				state_region = { add_state_trait = state_trait_city_of_rome }
			}
		every_state = {
			limit = {
				OR = {
					state_region = s:STATE_LOMBARDY
					state_region = s:STATE_ROMAGNA
					state_region = s:STATE_VENETIA
					}
				}
				state_region = { add_state_trait = state_trait_po_valley }
			}



	}

	option = {
		name = italy_tyrone_characters.1.a
		default_option = yes

		trigger_event = {
			id = italy_tyrone_characters.2
			popup = yes
		}
	}
}

# Treaty of Turin (France receives). Root = France. Sardinia (c:SAR) cedes Savoy/Provence.
# Only fire when the mod context is active (Sardinia has diplo stance JE), so it does not proc from vanilla/other triggers.
italyEventsTyrone.3 = {
	type = country_event
	placement = root

	title = italyEventsTyrone.3.t
	desc = italyEventsTyrone.3.d
	flavor = italyEventsTyrone.3.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"
	duration = 3

	trigger = {
		this = c:FRA
		exists = c:SAR
		c:SAR = { has_journal_entry = italy_diplo_stance }
	}

	option = {
		name = italyEventsTyrone.3.a
		# Sardinia gives Savoy and (if owned) Provence to France — base game pattern
		c:SAR ?= {
			ita_stance_increase_toward_france_effect = yes
		}
		trigger_event = { id = natural_borders.3 popup = yes }
	}
	option = {
		name = italyEventsTyrone.3.b
		c:SAR ?= {
			change_relations = { country = root value = -15 }
		}
	}
}

# Lombardy cession (Austria receives). Root = Austria. Proposer = country with italy_treaty_lombardy_proposer.
italyEventsTyrone.4 = {
	type = country_event
	placement = root

	title = italyEventsTyrone.4.t
	desc = italyEventsTyrone.4.d
	flavor = italyEventsTyrone.4.f

	event_image = {
		video = "ip3_official_reading_dictate"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"
	duration = 3

	trigger = { }

	option = {
		name = italyEventsTyrone.4.a
		# Whoever owns Lombardy (Austria or Austrian subject) cedes every Lombardy state to Sardinia
		any_country = {
			limit = {
				OR = {
					this = c:AUS
					is_subject_of = c:AUS
				}
				any_scope_state = {
					state_region = s:STATE_LOMBARDY
				}
			}
			any_scope_state = {
				limit = {
					state_region = s:STATE_LOMBARDY
				}
				set_state_owner = c:SAR
			}
		}
		c:SAR ?= {
			ita_stance_increase_toward_austria_effect = yes
			remove_variable = italy_treaty_lombardy_proposer
		}
		every_state = {
			limit = {
				state_region = s:STATE_LOMBARDY
			}
			set_state_owner = c:SAR
		}
	}
	option = {
		name = italyEventsTyrone.4.b
		c:SAR ?= {
			change_relations = { country = root value = -15 }
			remove_variable = italy_treaty_lombardy_proposer
		}
	}
}

		}
	}

# Laissez-faire enactment: shape the law (industrialists, partial control, or seize investment)
italyEventsTyrone.8 = {
	type = country_event
	placement = root

	title = italyEventsTyrone.8.t
	desc = italyEventsTyrone.8.d
	flavor = italyEventsTyrone.8.f

	gui_window = event_window_1char_tabloid
	left_icon = scope:industrialist_leader_scope

	category = enactment
	icon = "gfx/interface/icons/law_icons/laissez_faire.dds"
	duration = 3
	cooldown = { years = 30 }

	trigger = {
		is_enacting_law = law_type:law_laissez_faire
		NOT = {
			currently_enacting_law = {
				has_amendment = amendment_type:amendment_corporate_control
			}
		}
	}

	immediate = {
		currently_enacting_law = {
			save_scope_as = current_law_scope
		}
		ig:ig_industrialists ?= {
			save_scope_as = industrialists_ig
		}
		ig:ig_industrialists ?= {
			leader = {
				save_scope_as = industrialist_leader_scope
			}
		}
	}

	cancellation_trigger = {
		NOT = { currently_enacting_law = scope:current_law_scope }
	}

	option = {
		name = italyEventsTyrone.8.a
		currently_enacting_law = {
			add_amendment = {
				type = amendment_corporate_control
				sponsor = scope:industrialists_ig
				cooldown = 120
			}
		}
	}

	option = {
		name = italyEventsTyrone.8.b
		default_option = yes
		add_enactment_modifier = {
			name = laissez_faire_partial_control_stall
		}
	}

	option = {
		name = italyEventsTyrone.8.c
		add_modifier = {
			name = state_seized_investment
			years = 4
		}
	}
}
