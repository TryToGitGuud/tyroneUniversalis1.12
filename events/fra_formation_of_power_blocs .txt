fra_formation_of_power_blocs.1 = {
	type = country_event
	placement = root
	hidden = yes

	cooldown = { days = 1750 }

	trigger = {
		exists = c:FRA
		c:FRA = {
			is_ai = yes
			country_rank >= rank_value:unrecognized_major_power
			NOT = { has_law_or_variant = law_type:law_isolationism }
			NOT = { is_country_type = decentralized }
			is_revolutionary = no
			OR = {
				AND  = {			
					country_rank >= rank_value:unrecognized_major_power
					global_country_ranking <= 8
				}
				country_rank >= rank_value:great_power
			}	
			is_in_power_bloc = no
			is_at_war = no
			is_subject = no
			ruler = {
				AND = {
					NOT = {
						has_trait = trait_child
						has_trait = cautious
						has_trait = reserved
					}					
					NOT = {
						has_ideology = ideology:ideology_luddite
						has_ideology = ideology:ideology_protectionist
						has_ideology = ideology:ideology_vanguardist
						has_ideology = ideology:ideology_despotic_utopian
					}
				}
			}
			OR = {
				AND = {
					ig:ig_industrialists ?= { is_in_government = yes }
					has_law_or_variant = law_type:law_free_trade
				}
				AND = {
					has_strategy = ai_strategy_economic_imperialism
					has_law_or_variant = law_type:law_free_trade
				}
			}
			NOT = { has_journal_entry = je_divided_monarchists }
			NOT = {
				has_law_or_variant = law_type:law_anarchy
				has_law_or_variant = law_type:law_state_atheism
				has_law = law_type:law_technocracy
				has_law_or_variant = law_type:law_theocracy
				has_law = law_type:law_corporate_state
				has_radical_leftist_government = yes
			}
		}
		game_date >= 1837.1.1
	}

	immediate = {
		if = {
			limit = {
				exists = c:FRA
				c:FRA = {
					is_ai = yes
					country_rank >= rank_value:unrecognized_major_power
					NOT = { has_law_or_variant = law_type:law_isolationism }
					NOT = { is_country_type = decentralized }
					is_revolutionary = no
					OR = {
						AND  = {			
							country_rank >= rank_value:unrecognized_major_power
							global_country_ranking <= 8
						}
						country_rank >= rank_value:great_power
					}	
					is_in_power_bloc = no
					is_at_war = no
					is_subject = no
					ruler = {
						AND = {
							NOT = {
								has_trait = trait_child
								has_trait = cautious
								has_trait = reserved
							}							
							NOT = {
								has_ideology = ideology:ideology_luddite
								has_ideology = ideology:ideology_protectionist
								has_ideology = ideology:ideology_vanguardist
								has_ideology = ideology:ideology_despotic_utopian
							}
						}
					}
					OR = {
						AND = {
							ig:ig_industrialists ?= { is_in_government = yes }
							has_law_or_variant = law_type:law_free_trade
						}
						AND = {
							has_strategy = ai_strategy_economic_imperialism
							has_law_or_variant = law_type:law_free_trade
						}
					}
					NOT = { has_journal_entry = je_divided_monarchists }
					NOT = {
						has_law_or_variant = law_type:law_anarchy
						has_law_or_variant = law_type:law_state_atheism
						has_law = law_type:law_technocracy
						has_law_or_variant = law_type:law_theocracy
						has_law = law_type:law_corporate_state
						has_radical_leftist_government = yes
					}
				}
				game_date >= 1837.1.1
			}
			c:FRA = {
				create_power_bloc = {
					name = "Confédération du Grand Commerce Français"
					map_color = { 16 95 128 }
					identity = identity_trade_league
					principle = principle_external_trade_1
				}
			}
		}
	}
}
