				desc = germany_rework.1.d2
				trigger = {
					country_has_primary_culture = cu:south_german
				}
			}
		}
	}	
	flavor = {

			triggered_desc = {
				desc = germany_rework.1.f1
				trigger = {
					country_has_primary_culture = cu:north_german
				}
			}
			triggered_desc = {
				desc = germany_rework.1.f2
				trigger = {
					country_has_primary_culture = cu:south_german
				}
			}
		}
	}


	event_image = {

	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		set_variable = je_unification_fail
	}

	option = {
		name = germany_rework.1.a
		default_option = yes
		trigger = {
			c:PRU ?= THIS
		}
		add_modifier = {
			name = brothers_war_defeat_pru
			days = long_modifier_time
			is_decaying = yes
		}
	}
	option = {
		name = germany_rework.1.b
		default_option = yes
		trigger = {
			c:AUS ?= THIS
			has_law_or_variant = law_type:law_monarchy
			NOT = {
				has_variable = empire_of_austria_variable
			}
		}
		add_modifier = {
			name = brothers_war_defeat_aus
			days = long_modifier_time
			is_decaying = yes
		}
		add_radicals = {
			value = large_radicals
		}
		save_scope_as = austria_scope
		s:STATE_CENTRAL_HUNGARY = {
			random_scope_state = {
				save_scope_as = budapest_state_scope
			}
		}
		custom_tooltip = {
			text = matter_of_hungary_ausgleich_tt
			if = {
				limit = {
					NOT = {
						exists = c:HUN
					}
				}
				create_country = {
					origin = root
					tag = HUN
					state = scope:budapest_state_scope
					on_created = {
					}
				}
			}
		}
		trigger_event = { id = austria_events.4 popup = yes}
		set_variable = established_austria_hungary_var
	}
	option = {
		name = germany_rework.1.c
		default_option = yes
		trigger = {
			c:AUS ?= THIS
		}
		if = {
			limit = {
				NOT = {
					has_variable = finished_hungary_var
				}
			}
			add_modifier = {
				name = brothers_war_defeat_aus
				days = long_modifier_time
				is_decaying = yes
			}
		}
		add_radicals = {
			value = large_radicals
			culture = cu:hungarian
		}
		save_scope_as = austria_scope
		s:STATE_CENTRAL_HUNGARY = {
			random_scope_state = {
				save_scope_as = budapest_state_scope
			}
		}
		custom_tooltip = {
			text = hungary_lead_to_war_tt
			if = {
				limit = {
					NOT = {
						exists = c:HUN
					}
				}
				create_country = {
					origin = root
					tag = HUN
					state = scope:budapest_state_scope
				}
				every_state = {
					limit = {
						OR = {
							owner ?= ROOT
							owner = {
								is_subject_of = ROOT
							}
						}
						NOT = {
							c:HUN ?= owner
						}
						state_region = {
							is_homeland = cu:hungarian
						}
					}
					set_state_owner = c:HUN
					set_state_type = incorporated
				}
			}
			if = {
				limit = {
					exists = c:HUN
				}
				trigger_event = {
					id = austria_events.6
				}
			}
		}
	}
}

germany_rework.2 = {
	type = country_event
	placement = ROOT

	title = germany_rework.2.t
	desc = germany_rework.2.d
	flavor = germany_rework.2.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		every_country = {
			limit = {
				country_rank < rank_value:great_power
				OR = {
					AND = {
						root = { country_has_primary_culture = cu:north_german }
						country_has_primary_culture = cu:north_german
					}
					AND = {
						root = { country_has_primary_culture = cu:south_german }
						country_has_primary_culture = cu:south_german
					}
				}
				NOR = {
					THIS = ROOT
					is_subject_type = subject_type_colony
					is_subject_type = subject_type_dominion
					is_subject_type = subject_type_chartered_company
				}
			}

			if = {
				limit = {
					is_ai = yes
				}
				root = { annex_with_incorporation = prev }
			}
			else = {
				trigger_event = { id = german_unification.2 days = 0 popup = yes }
			}
		}
	}

	option = {
		name = germany_rework.2.a
		default_option = yes
		if = {
			limit = {
				country_has_primary_culture = cu:north_german
				NOT = { exists = c:NGF }
				OR = {
					country_tier = kingdom
					country_tier = principality
				}
			}

			# Reconcile with Austria if we didn't conquer states from them or release any crucial territories
			hidden_effect = {
				if = {
					limit = {
						exists = c:AUS
						c:AUS ?= {
							country_rank >= rank_value:great_power
							NOT = { has_claim = root }
							owns_entire_state_region = STATE_BOHEMIA
							owns_entire_state_region = STATE_CENTRAL_HUNGARY
							owns_entire_state_region = STATE_LOMBARDY
						}
					}

					c:AUS ?= {
						set_mutual_secret_goal = {
							country = root
							secret_goal = reconcile
						}
					}
				}
			}
			if = {
				limit = {
					c:PRU ?= THIS
				}
				annex = c:HOH
			}		
				
			change_tag = NGF
			custom_tooltip = {
				text = get_claims_germany_tt
				give_claims_on_germany = yes
			}
		}
	}
}

germany_rework.3 = {
    type = country_event
    placement = root

    title = germany_rework.3.t
    desc = germany_rework.3.d
    flavor = germany_rework.3.f

    event_image = {
		video = "europenorthamerica_political_extremism"
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

    icon = "gfx/interface/icons/event_icons/crown_in_the_gutter.dds"

    duration = 3

	trigger = {
		is_at_war = no
		is_diplomatic_play_committed_participant = no	
		OR = {
			has_technology_researched = pan-nationalism
			game_date >= 1848.1.1
			has_journal_entry = je_springtime_of_the_peoples		
		}	
	}	

    option = {
        name = germany_rework.3.a
		set_variable = accepted_gutter_crown
		add_radicals = {
			interest_group = ig:ig_armed_forces
			value = 0.5
		}
		add_radicals = {
			interest_group = ig:ig_devout
			value = 0.5
		}
		add_radicals = {
			interest_group = ig:ig_landowners
			value = 0.5
		}
		add_modifier = {
			name = gutter_germany
			days = long_modifier_time
		}
		every_country = {
			limit = {
				country_rank < rank_value:great_power
				OR = {
					country_has_primary_culture = cu:north_german
					country_has_primary_culture = cu:south_german
				}
				NOR = {
					THIS = ROOT
					THIS = c:AUS
					THIS = c:LUX
					THIS = c:TRS
					THIS = c:BNT
					is_subject_type = subject_type_colony
					is_subject_type = subject_type_dominion
					is_subject_type = subject_type_chartered_company
				}
			}

			if = {
				limit = {
					is_ai = yes
				}
				root = { annex_with_incorporation = prev }
			}
		}
		if = {
            limit = {
                NOT = {
                    has_law = law_type:law_census_voting
                }
            }
            activate_law = law_type:law_census_voting
        }
        if = {
            limit = {
                NOT = {
                    has_law = law_type:law_right_of_assembly
                }
            }
            activate_law = law_type:law_right_of_assembly
        }
        if = {
            limit = {
                NOT = {
                    has_law = law_type:law_national_guard
                }
            }
            activate_law = law_type:law_national_guard
        }
		change_tag = GER
        add_primary_culture = cu:south_german
		c:AUS = {
			trigger_event = { id = germany_rework.4 popup = yes }
		}
    }

    option = {
        name = germany_rework.3.b
		set_variable = accepted_gutter_crown
		add_modifier = {
			name = gutter_german_republic
			days = long_modifier_time
		}
		add_radicals = {
			interest_group = ig:ig_armed_forces
			value = 0.9
		}
		add_radicals = {
			interest_group = ig:ig_devout
			value = 0.9
		}
		add_radicals = {
			interest_group = ig:ig_landowners
			value = 0.9
		}
		every_country = {
			limit = {
				country_rank < rank_value:great_power
				OR = {
					country_has_primary_culture = cu:north_german
					country_has_primary_culture = cu:south_german
				}
				NOR = {
					THIS = ROOT
					THIS = c:AUS
					THIS = c:LUX
					THIS = c:TRS
					THIS = c:BNT
					is_subject_type = subject_type_colony
					is_subject_type = subject_type_dominion
					is_subject_type = subject_type_chartered_company
				}
			}

			if = {
				limit = {
					is_ai = yes
				}
				root = { annex_with_incorporation = prev }
			}
		}
		if = {
            limit = {
                NOR = {
                    has_law = law_type:law_presidential_republic
                    has_law = law_type:law_parliamentary_republic
                }
            }
			activate_law = law_type:law_presidential_republic
        }
		if = {
            limit = {
                NOT = {
                    has_law = law_type:law_universal_suffrage
                }
            }
            activate_law = law_type:law_universal_suffrage
        }
        if = {
            limit = {
                NOT = {
					has_law = law_type:law_protected_speech
                }
            }
            activate_law = law_type:law_protected_speech
        }
        if = {
            limit = {
                NOT = {
                    has_law = law_type:law_national_guard
                }
            }
            activate_law = law_type:law_national_guard
        }
		change_tag = GER
        add_primary_culture = cu:south_german
		c:AUS = {
			trigger_event = { id = germany_rework.4 popup = yes }
		}
    }

	option = {
        name = germany_rework.3.c
        default_option = yes
		set_variable = declined_gutter_crown
		add_modifier = {
			name = rejected_gutter
			days = long_modifier_time
		}
    }
}
