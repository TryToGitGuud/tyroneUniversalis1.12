	parent = law_mutualist_economy

	allowed_laws = {
		law_mutualist_economy
	}

	modifier = {
		country_authority_mult = 0.10
		country_government_dividends_reinvestment_add = -0.20
		country_government_dividends_efficiency_add = 0.40
	}

	possible = {
		has_variable = proudhon_var
		has_law = law_type:law_mutualist_economy
	}

	would_sponsor = {
		OR = {
			is_interest_group_type = ig_trade_unions
			leader ?= {
				has_ideology = ideology:ideology_mutualist
			}
		}
	}

	ai_will_revoke = {
		always = no
	}

	can_repeal = {
		legitimacy >= legitimacy_to_repeal_amendment
	}

	amendment_activism_multiplier = 0.5
}
