# Amendment granted when completing the Austria military JE the hard way (National Guard + Home Affairs level 2).
# Applied to National Guard law: +20% training rate, +5% home defense.
amendment_italian_gendarme = {

	parent = law_national_guard

	allowed_laws = {
		law_national_guard
	}

	modifier = {
		building_training_rate_mult = 0.20
		battle_defense_owned_province_mult = 0.05
	}

	possible = {
		has_variable = ita_austria_military_done
		NOT = { has_variable = ita_austria_military_complete_clicked }
	}

	would_sponsor = {
		always = no
	}

	can_repeal = {
		legitimacy >= legitimacy_to_repeal_amendment
	}
}
