movement_internationale = {
	category = movement_category_ideological

	active_law_radicalism_multiplier = 0.75

	ideology = ideology_internationale_marxist_movement
	character_ideologies = {
		ideology_anarchist
		ideology_mutualist
		ideology_syndicalist
		ideology_blanquist
		ideology_marxist
	}

	creation_trigger = {
		OR = {
			has_variable = bakunin_var
			has_variable = proudhon_var
			has_variable = bel_de_paepe_var
			has_variable = blanqui_var
			has_variable = karl_marx_var
		}
	}
	creation_weight = {
		value = 0
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}

	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_anarchist
			has_ideology = ideology:ideology_mutualist
			has_ideology = ideology:ideology_syndicalist
			has_ideology = ideology:ideology_blanquist
			has_ideology = ideology:ideology_marxist
			AND = {
				is_interest_group_type = ig_trade_unions
				has_ideology = ideology:ideology_humanitarian
			}
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_anarchist
					has_ideology = ideology:ideology_mutualist
					has_ideology = ideology:ideology_syndicalist
					has_ideology = ideology:ideology_blanquist
					has_ideology = ideology:ideology_marxist
				}
			}
			multiply = 5
		}
	}

	pop_support_trigger = {
		NOR = {
			is_pop_type = capitalists
			is_pop_type = aristocrats
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_high_literacy
		movement_support_below_expected_sol
		movement_support_radicalization
		movement_support_academics
		movement_support_laborers
		movement_support_machinists
		movement_support_soldiers_conscripts
	}

	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
					OR = {
						is_being_suppressed = yes
						is_being_bolstered = yes
					}
				}
			}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}

		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}

		if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = laborers
				}
				add = {
					value = 18
					desc = "POP_LABORERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = machinists
				}
				add = {
					value = 25
					desc = "POP_MACHINISTS"
				}
			}
			else = {
				add = {
					value = 6
					desc = "LOWER_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			multiply = {
				value = 0.10
				desc = "MIDDLE_NO_ICON"
			}
			if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 3
					desc = "POP_ACADEMICS"
				}
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
				strata = lower
			}
			add = {
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}

		multiply = {
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = {
			desc = "POP_RADICALS"
			value = pop_radical_fraction
			divide = 2
			add = 1
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}

		weight = {
			value = 100
		}

		state_weight = {
			value = state_urbanization_rate
			add = scope:clout
			add = { value = this.turmoil divide = 2 }
			subtract = this.loyalists

			if = {
				limit = { is_capital = yes }
				multiply = 10.0
			}
		}
	}

	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}

		weight = {
			value = 1000
		}

		state_weight = {
			value = state_urbanization_rate
			add = { value = this.turmoil divide = 2 }
			subtract = this.loyalists

			if = {
				limit = { is_capital = yes }
				multiply = 10.0
			}
		}
	}
}
