	category = movement_category_cultural

	ideology = ideology_pluralist_movement_1
	character_ideologies = {
		ideology_reformer
	}

	can_cause_obstinance = yes

	creation_trigger = {}
	creation_weight = {
		value = 100
		if = {
			limit = {
				owner ?= {
					has_technology_researched = nationalism
				}
			}
			multiply = {
				value = 2
			}
		}
		if = {
			limit = {
				owner ?= {
					has_variable = failed_tanzimat
				}
			}
			multiply = {
				value = 2
			}
		}
		if = {
			limit = {
				owner ?= {
					has_journal_entry = je_springtime_of_the_peoples
				}
			}
			multiply = {
				value = 2
			}
		}
		if = {
			limit = {
				owner ?= {
					has_journal_entry = je_great_eastern_crisis
				}
			}
			multiply = {
				value = 2
			}
		}
	}
	on_created = {
		if = {
			limit = {
				culture = {
					AND = {
						shares_heritage_trait_with_any_primary_culture = root.owner
						OR = {
							shares_language_trait_group_with_any_primary_culture = root.owner
							shares_tradition_trait_with_any_primary_culture = root.owner
						}
					}
				}
			}
			set_core_ideology = ideology_pluralist_movement_1
		}
		else_if = {
			limit = {
				culture = {
					shares_heritage_trait_group_with_any_primary_culture = root.owner
					OR = {
						shares_language_trait_group_with_any_primary_culture = root.owner
						shares_tradition_trait_with_any_primary_culture = root.owner
					}
				}
			}
			set_core_ideology = ideology_pluralist_movement_2
		}
		else_if = {
			limit = {
				culture = {
					OR = {
						shares_heritage_trait_group_with_any_primary_culture = root.owner
						shares_language_trait_with_any_primary_culture = root.owner
					}
				}
			}
			set_core_ideology = ideology_pluralist_movement_3
		}
		else = {
			set_core_ideology = ideology_pluralist_movement_4
		}
		if = {
			limit = {
				owner = {
					has_variable_list = failed_uprisings_culture_list
					is_target_in_variable_list = { name = failed_uprisings_culture_list target = root.culture }
				}
			}
			owner = {
				remove_list_variable = { name = failed_uprisings_culture_list target = root.culture }
			}
			add_modifier = {
				name = movement_suppressed_after_civil_war_copy
				days = very_long_modifier_time
				is_decaying = yes
			}
		}
	}

	disband_trigger = {
		scope:political_movement ?= {
			OR = {
				culture = {
					is_primary_culture_of = root
				}
				root = {
					AND = {
						NOT = {
							any_scope_state = {
								state_region = {
									is_homeland = scope:political_movement.culture
								}
								culture_percent_state = {
									target = scope:political_movement.culture
									value >= 0.18
								}
							}
						}
						OR = {
							culture_percent_country = {
								target = scope:political_movement.culture
								value <= 0.08
							}
							cultural_acceptance_base = { target = scope:political_movement.culture value > acceptance_status_2 }
						}
					}
				}
			}
		}
	}

	culture_selection_trigger = {
		NOT = { is_primary_culture_of = scope:country }
		OR = {
			culture_current_fervor >= 10
			scope:country = {
				any_scope_diplomatic_pact = {
					is_diplomatic_action_type = support_separatism
					second_country = prev
					"diplomatic_pact_other_country(scope:country)" = {
						country_has_primary_culture = root
					}
				}
			}
		}
		scope:country = {
			OR = {
				any_scope_state = {
					state_region = {
						is_homeland = root
					}
					culture_percent_state = {
						target = root
						value >= 0.2
					}
				}
				AND = {
					culture_percent_country = {
						target = root
						value >= 0.1
					}
					cultural_acceptance_base = { target = root value <= acceptance_status_2 }
				}
			}
		}
	}
	culture_selection_weight = {
		value = 100

		if = {
			limit = {
				scope:country = {
					any_scope_diplomatic_pact = {
						is_diplomatic_action_type = support_separatism
						second_country = prev
						"diplomatic_pact_other_country(scope:country)" = {
							country_has_primary_culture = root
						}
					}
				}
			}
			multiply = 3
		}
	}

	character_support_trigger = {
		OR = {
			culture = scope:culture
			has_ideology = ideology:ideology_humanitarian
			has_ideology = ideology:ideology_humanitarian_royalist
			has_ideology = ideology:ideology_sovereignist_leader
		}
		NOT = {
			has_variable = anti_nationalist_var # Please stop becoming a Polish nationalist, Rosa.
		}
	}
	character_support_weight = {
		value = 50
		if = {
			limit = {
				has_ideology = ideology:ideology_sovereignist_leader
			}
			multiply = {
				value = 10
			}
		}
		else_if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_humanitarian_royalist
					has_ideology = ideology:ideology_humanitarian
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_socialist_ideology = yes
			}
			multiply = {
				value = 0.5
			}
		}
	}

	pop_support_trigger = {
		culture = scope:culture
	}

	pop_support_factors = {
		movement_support_national_awakening
		movement_support_cultural_fervor
		movement_support_cultural_discrimination
		movement_support_je_sick_man_main
		movement_support_je_matter_of_hungary
		movement_support_aristocrats_polish
		movement_support_academics
		movement_support_clergymen
		movement_support_clerks
		movement_support_soldiers

		movement_support_pan_nationalism_tech
		movement_support_support_separatism_action

			limit = {
				exists = scope:political_movement

					OR = {
						is_being_suppressed = yes
						is_being_bolstered = yes
					}
				}
			}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}

		if = {
			limit = {
				strata = upper
			}
			if = {
				limit = {
					is_pop_type = aristocrats
				}
				if = {
					limit = {
						culture = cu:polish
						state ?= {
							OR = {
								has_variable = lingering_plc_resistance
								has_variable = christ_of_nations_target
							}
						}
					}
					add = {
						value = 50
						desc = "POP_POLISH_ARISTOCRATS"
					}
				}
				else_if = {
					limit = {
						culture = cu:hungarian
						owner ?= {
							has_journal_entry = je_matter_of_hungary
						}
					}
					add = {
						value = 50
						desc = "je_matter_of_hungary"
					}
				}
				else = {
					add = {
						value = 12
						desc = "POP_ARISTOCRATS"
					}
				}
			}
			else = {
				add = {
					value = 18
					desc = "UPPER_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 40
					desc = "POP_ACADEMICS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = clergymen
				}
				if = {
					limit = {
						NOT = {
							this.religion = owner.religion
						}
					}
					add = {
						value = 40
						desc = "POP_CLERGYMEN"
					}
				}
				else = {
					add = {
						value = 15
						desc = "POP_CLERGYMEN"
					}
				}
			}
			if = {
				limit = {
					is_pop_type = shopkeepers
				}
				add = {
					value = 25
					desc = "POP_SHOPKEEPERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 20
					desc = "POP_OFFICERS"
				}
			}
			else = {
				add = {
					value = 15
					desc = "MIDDLE_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 25
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 15
					desc = "POP_SOLDIERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = peasants
				}
				add = {
					value = 5
					desc = "POP_PEASANTS"
				}
			}
			else = {
				add = {
					value = 10
					desc = "LOWER_NO_ICON"
				}
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
			}
			add = { # Increases likelihood of support if impoverished
				desc = "POP_POVERTY"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}

		if = {
			limit = {
				state = {
					state_cultural_acceptance = {
						target = PREV.culture
						value < acceptance_status_5
					}
				}
			}
			add = { # From +2 at 80 to +10 at 0
				desc = "POP_DISCRIMINATED"
				value = 100
				subtract = pop_acceptance
				divide = 10
			}
		}
		else = { # Non-discriminated pops are less likely to form separatist movements
			multiply = {
				desc = "POP_ACCEPTANCE"
				value = 0.25
			}
		}

		multiply = { # Multiplied by 2x max
			desc = "POP_RADICALS"
			value = pop_radical_fraction
			add = 1
		}

		if = {
			limit = {
				owner ?= {
					has_technology_researched = pan-nationalism
				}
			}
			multiply = {
				value = 1.5
				desc = "PAN_NATIONALISM_TECH"
			}
		}
		else_if = {
			limit = {
				owner ?= {
					has_technology_researched = nationalism
				}
			}
			multiply = {
				value = 1.25
				desc = "NATIONALISM_TECH"
			}
		}

		multiply = { # Range from x0.775 to 2.275
			desc = "POP_CULTURE_FERVOR"
			value = culture.culture_current_fervor
			subtract = 15
			multiply = 0.015
			add = 1
		}

		if = { # Funnel people into Indias unique movements
			limit = {
				owner ?= {
					c:BIC ?= this
				}
				culture = { has_discrimination_trait_group = heritage_group_south_asian }
			}
			multiply = 0.5

			if = {
				limit = {
					owner ?= {
						has_technology_researched = pan-nationalism
					}
				}
				multiply = 0.1
			}
		}

		if = {
			limit = {
				owner ?= {
					has_variable = failed_tanzimat
				}
				culture = { has_discrimination_trait_group = heritage_group_european }
			}
			multiply = {
				value = 1.25
				desc = "FAILED_TANZIMAT_REFORMS"
			}
		}

		if = {
			limit = {
				owner ?= {
					has_journal_entry = je_matter_of_hungary
				}
				culture = cu:hungarian
			}
			multiply = {
				value = 1.25
				desc = "je_matter_of_hungary"
			}
		}

		if = {
			limit = {
				culture ?= {
					culture_has_national_awakening = yes
				}
			}
			multiply = {
				value = 1.5
				desc = "POP_ACTIVE_NATIONAL_AWAKENING"
			}
		}

		if = {
			limit = {
				owner = {
					any_scope_diplomatic_pact = {
						is_diplomatic_action_type = support_separatism
						second_country = prev
						"diplomatic_pact_other_country(root.owner)" = {
							country_has_primary_culture = root.culture
						}
					}
				}
			}
			multiply = {
				desc = "SUPPORT_SEPARATISM_ACTION"
				value = support_separatism_total_pop_support
				multiply = 0.01
				add = 1
			}
		}
	}

	secession = {
		possible = {
			owner ?= {
				any_scope_state = {
					state_region = { is_homeland = root.culture }
					culture_percent_state = {
						target = root.culture
						value >= 0.05
					}
				}
			}
		}

		weight = {
			value = 1000
		}

		state_weight = {
			value = 0

			if = {
				limit = {
					is_capital = no
					state_region = { is_homeland = scope:political_movement.culture }
					culture_percent_state = {
						target = scope:political_movement.culture
						value >= 0.05
					}
				}
				add = { # These three values must add up to more than 1 - 5 x Obstinance, 1/2 Turmoil, percent of pops of this culture
					value = -1
					add = "this.culture_percent_state(scope:political_movement.culture)"
					add = {
						value = this.turmoil
						multiply = 0.5
						}
					add = {
						value = this.obstinance
						multiply = 5
					}
					multiply = 100
				}
			}
		}

		interest_group_can_join = {
			trigger_if = {
				limit = {
					c:BIC ?= owner
				}
				is_indian_interest_group = yes
			}
			trigger_else_if = {
				limit = {
					owner = {
						country_is_colonial_or_company = yes
					}
				}
				NOT = {
					has_ideology = ideology:ideology_colonialist
				}
			}
			trigger_else = { # For example - the Austrian Aristocracy can only secede with a Polish uprising if the leader of the IG is Polish
				leader = { culture = scope:political_movement.culture }
			}
		}

		target_fraction_of_states = {
			value = political_movement_identity_support
			add = scope:clout
			multiply = 3
			min = 0.15
			max = 0.75
		}
	}

	additional_radicalism_factors = {
		add = {
			value = "owner.average_sol"
			subtract = "owner.average_sol_for_culture(scope:culture)"
			multiply = 0.05
			max = 0.5
			min = 0
			desc = "STANDARD_OF_LIVING_FOR_CULTURE"
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_subjecthood
				}
			}
			add = {
				value = 100
				subtract = "owner.country_average_cultural_acceptance(scope:culture)"
				divide = 200 # To get this on the level of the other laws
				subtract = 0.10 # Compensate for religious acceptance, which is currently inaccessible from here
				desc = "DISCRIMINATION_LEVEL_subjecthood"
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_multiculturalism
				}
			}
			subtract = {
				value = 25
			}
		}

		else = {
			if = {
				limit = {
				owner = { cultural_acceptance_base = { target = scope:culture value <= acceptance_status_1 } }
			}

			add = {
				value = 0.3
				desc = "DISCRIMINATION_LEVEL_violent_hostility"
			}
		}
		else_if = {
			limit = {
				owner = { cultural_acceptance_base = { target = scope:culture value <= acceptance_status_2 } }
			}

			add = {
				value = 0.2
				desc = "DISCRIMINATION_LEVEL_cultural_erasure"
			}
		}
		else_if = {
			limit = {
				owner = { cultural_acceptance_base = { target = scope:culture value <= acceptance_status_3 } }
			}

			add = {
				value = 0.1
				desc = "DISCRIMINATION_LEVEL_open_prejudice"
			}
		}
		}

		if = {
			limit = {
				owner = {
					any_scope_diplomatic_pact = {
						is_diplomatic_action_type = support_separatism
						second_country = prev
						"diplomatic_pact_other_country(root.owner)" = {
							country_has_primary_culture = root.culture
						}
					}
				}
			}
			add = {
				desc = "SUPPORT_SEPARATISM_ACTION"
				value = support_separatism_radicalism
			}
		}

		subtract = {
			value = 0.25
			subtract = political_movement_identity_support
			min = 0
			desc = "LOW_MOVEMENT_IDENTITY_SUPPORT"
		}

		if = {
			limit = {
				owner = { cultural_acceptance_base = { target = scope:culture value <= acceptance_status_4 } }
			}
			add = {
				desc = "FERVOR_OF_CULTURE_ACTIVISM"
				value = scope:culture.culture_current_fervor
				divide = 400
			}
		}

		if = {
			limit = {
				owner = {
					has_modifier = modifier_polish_organic_statute
				}
			}
			add = {
				desc = "modifier_polish_organic_statute"
				value = 0.15
			}
		}

		if = {
			limit = {
				owner = {
					any_neighbouring_state = {
						owner = {
							NOT = { THIS = ROOT.owner }
							any_political_movement = {
								culture ?= ROOT.culture
								is_insurrectionary = yes
							}
						}
					}
				}
			}
			add = {
				desc = "NEIGHBOURING_COUNTRY_WITH_CULTURE_REVOLUTION"
				value = 0.15
			}
		}

		if = {
			limit = {
				owner = {
					any_political_movement = {
						NOT = { this = root } # Insurrectionary movements should not make themselves more insurrectionary
						is_insurrectionary = yes
					}
				}
			}
			add = {
				desc = "SIMULTANEOUS_INSURRECTIONARY_MOVEMENTS"
				owner = {
					every_political_movement = {
						limit = {
							NOT = { this = root } # Insurrectionary movements should not make themselves more insurrectionary
							is_insurrectionary = yes
						}
						if = {
							limit = {
								exists = this.political_movement_identity_support
							}
							add = { # Identity support can get a bit high
								value = this.political_movement_identity_support
								multiply = 0.5
							}
						}
						else = {
							add = this.political_movement_support
						}
					}
				}
				min = 0.05
				max = 0.25
			}
		}
	}
}

REPLACE:movement_fascist = {
	category = movement_category_ideological

	active_law_radicalism_multiplier = 1

	ideology = ideology_fascist_movement
	character_ideologies = {
		ideology_fascist
		ideology_integralist
		ideology_corporatist_leader
	}

	creation_trigger = {
		has_technology_researched = corporatism
		country_should_exclusively_focus_on_modernization = no
	}
	creation_weight = {
		value = 100
		if = {
			limit = {
				owner ?= { # If a movement of this type exists in a neighbouring country, make it more likely to spawn here as well
					any_neighbouring_state = {
						owner = {
							prestige > ROOT.owner.prestige
							any_political_movement = {
								is_political_movement_type = movement_fascist
							}
						}
					}
				}
			}
			multiply = {
				value = 2
			}
		}
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
		if = {
			limit = {
				owner ?= {
					any_primary_culture = {
						has_discrimination_trait = language_lusophone
					}
				}
			}
			set_core_ideology = ideology_integralist_movement
		}
	}

	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_fascist
			has_ideology = ideology:ideology_integralist
			has_ideology = ideology:ideology_corporatist_leader
			has_ideology = ideology:ideology_ethno_nationalist
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_fascist
					has_ideology = ideology:ideology_integralist
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_corporatist_leader
					has_ideology = ideology:ideology_ethno_nationalist
				}
			}
			multiply = {
				value = 2
			}
		}
	}

	pop_support_trigger = {
		NOT = {
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_literacy
		movement_support_middle_below_expected_sol
		movement_support_radicalization
		movement_support_unemployed
		movement_support_capitalists
		movement_support_aristocrats
		movement_support_shopkeepers
		movement_support_officers
		movement_support_farmers
		movement_support_clerks
		movement_support_soldiers
		movement_support_number_of_claims
	}

	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
						OR = {
							is_being_suppressed = yes
							is_being_bolstered = yes
						}
					}
				}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}

		if = {
			limit = {
				strata = upper
			}
			if = {
				limit = {
					is_pop_type = capitalists
				}
				add = {
					value = 12
					desc = "POP_CAPITALISTS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = aristocrats
				}
				add = {
					value = 5
					desc = "POP_ARISTOCRATS" # Fascism is too populist
				}
			}
			else = {
				add = {
					value = 5
					desc = "UPPER_NO_ICON"
				}
			}

			if = {
				limit = {
					owner ?= {
						any_political_movement = {
							OR = {
								is_political_movement_type = movement_socialist
								is_political_movement_type = movement_anarchist
								is_political_movement_type = movement_communist
								is_political_movement_type = movement_labor
								is_political_movement_type = movement_nihilist
							}
							political_movement_support >= 0.05
						}
					}
				}
				multiply = {
					desc = "STRONG_SOCIALIST_MOVEMENT"
					value = 1.5
					owner ?= {
						every_political_movement = {
							limit = {
								OR = {
									is_political_movement_type = movement_socialist
									is_political_movement_type = movement_anarchist
									is_political_movement_type = movement_communist
									is_political_movement_type = movement_labor
									is_political_movement_type = movement_nihilist
								}
							}
							add = this.political_movement_radicalism
						}
					}
				}
			}
			if = {
				limit = {
					owner = {
						number_of_claims > 0
					}
				}

				add = {
					value = culture.culture_current_fervor # Add a base value for having claims at all
					divide = 100 # Culture is 1-100
					multiply = {
						value = owner.number_of_claims
					}
					max = 10
					desc = "NUMBER_OF_CLAIMS"
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = shopkeepers
				}
				add = {
					value = 15
					desc = "POP_SHOPKEEPERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = bureaucrats
				}
				add = {
					value = 8
					desc = "POP_BUREAUCRATS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = officers
				}
				add = {
					value = 8
					desc = "POP_OFFICERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = farmers
				}
				add = {
					value = 5
					desc = "POP_FARMERS"
				}
			}
			else = {
				add = {
					value = 4
					desc = "MIDDLE_NO_ICON"
				}
			}

			if = {
				limit = {
					owner ?= {
						any_political_movement = {
							OR = {
								is_political_movement_type = movement_socialist
								is_political_movement_type = movement_anarchist
								is_political_movement_type = movement_communist
								is_political_movement_type = movement_labor
								is_political_movement_type = movement_nihilist
							}
							political_movement_support >= 0.05
						}
					}
				}
				multiply = {
					desc = "STRONG_SOCIALIST_MOVEMENT"
					value = 1
					owner ?= {
						every_political_movement = {
							limit = {
								OR = {
									is_political_movement_type = movement_socialist
									is_political_movement_type = movement_anarchist
									is_political_movement_type = movement_communist
									is_political_movement_type = movement_labor
									is_political_movement_type = movement_nihilist
								}
							}
							add = this.political_movement_radicalism
						}
					}
				}
			}
			if = {
				limit = {
					owner = {
						number_of_claims > 0
					}
				}
				add = {
					value = culture.culture_current_fervor # Add a base value for having claims at all
					divide = 100 # Culture is 1-100
					multiply = {
						value = owner.number_of_claims
					}
					max = 10
					desc = "NUMBER_OF_CLAIMS"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 5
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 3
					desc = "POP_SOLDIERS"
				}
			}
			else = {
				add = {
					value = 2
					desc = "LOWER_NO_ICON"
				}
			}
			if = {
				limit = {
					OR = {
						is_pop_type = clerks
						is_pop_type = soldiers
					}
					owner = {
						number_of_claims > 0
					}
				}
				add = {
					value = culture.culture_current_fervor # Add a base value for having claims at all
					divide = 125 # Culture is 1-100
					multiply = {
						value = owner.number_of_claims
					}
					max = 5
					desc = "NUMBER_OF_CLAIMS"
				}
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
				OR = {
					strata = middle
					strata = upper
				}
			}
			add = { # Increases likelihood of support if impoverished
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}
		else_if = {
			limit = {
				standard_of_living > state.average_expected_sol
				OR = {
					strata = middle
					strata = upper
				}
			}
			add = { # Decreases likelihood of support if well-off
				desc = "IS_ABOVE_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				min = -5
			}
		}

		if = {
			limit = {
				is_employed = no
			}
			multiply = {
				value = 1.5
				desc = "UNEMPLOYED"
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = {
			value = 0.011
			multiply = pop_acceptance
			desc = "ACCEPTANCE"
		}

		if = {
			limit = {
				OR = {
					strata = upper
					strata = middle
					is_pop_type = clerks
				}
			}
			multiply = { # Multiplied by 2x max
				desc = "POP_RADICALS"
				value = pop_radical_fraction
				multiply = 2
				add = 1
			}
		}
		else = {
			multiply = { # Multiplied by 1.5x max
				desc = "POP_RADICALS"
				value = pop_radical_fraction
				divide = 2
				add = 1
			}
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}

		weight = {
			value = 100
		}

		state_weight = {
			value = 1
		}
	}

	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}

		weight = {
			value = 1000
		}
	}

	additional_radicalism_factors = {

		if = {
			limit = {
				owner = {
					any_neighbouring_state = {
						owner = {
							NOT = { THIS = ROOT.owner }
							any_political_movement = {
								is_political_movement_type = movement_fascist
								is_insurrectionary = yes
							}
						}
					}
				}
			}
			add = {
				desc = "NEIGHBOURING_COUNTRY_WITH_REVOLUTION"
				value = 0.15
			}
		}
	}
}
