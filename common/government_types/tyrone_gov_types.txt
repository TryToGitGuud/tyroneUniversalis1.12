REPLACE:gov_dual_monarchy = {
	transfer_of_power = hereditary

	male_ruler = "RULER_TITLE_EMPEROR_KING"
	female_ruler = "RULER_TITLE_EMPRESS_QUEEN"

	male_heir = "RULER_TITLE_ARCHDUKE"
	female_heir = "RULER_TITLE_ARCHDESS"

	possible = {
		has_law = law_type:law_monarchy
		OR = {
			AND = {
				has_variable = cisleithania_var
				any_subject_or_below = {
					is_subject_type = subject_type_personal_union
					OR = {
						has_variable = transleithania_var
						has_variable = triune_kingdom_var
					}
					count = 1
				}
			}
			AND = {
				is_subject_type = subject_type_personal_union
				has_variable = transleithania_var
				any_overlord_or_above = {
					has_variable = cisleithania_var
					NOT = {
						any_subject_or_below = {
							has_variable = triune_kingdom_var
						}
					}
				}
			}
			AND = {
				is_subject_type = subject_type_personal_union
				has_variable = triune_kingdom_var
				any_overlord_or_above = {
					has_variable = cisleithania_var
					NOT = {
						any_subject_or_below = {
							has_variable = transleithania_var
						}
					}
				}
			}
			has_variable = federated_hungarians_var
		}
		has_gov_regency = no
		NOT = {
			has_variable = empire_of_austria_variable
		}
	}

	on_government_type_change = {
		change_to_hereditary = yes
	}
	on_post_government_type_change = {
		post_change_to_hereditary = yes
	}
}
