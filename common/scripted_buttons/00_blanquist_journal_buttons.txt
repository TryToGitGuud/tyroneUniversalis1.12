je_blanquism_prepare_cells = {
	name = je_blanquism_prepare_cells
	desc = je_blanquism_prepare_cells_desc

	visible = {
		has_journal_entry = je_blanquism
		NOT = { has_variable = blanquist_cells_prepared }
	}

	possible = {
		NOT = {
			custom_tooltip = {
				text = blanquist_cooldown_tt
				has_variable = blanquist_interaction_cooldown
			}
		}
		
		any_political_movement = {
			is_political_movement_type = movement_internationale
			has_core_ideology = ideology:ideology_internationale_blanquist_movement
			political_movement_support >= 0.10
		}
		radical_fraction = {
			value >= 0.10
		}
	}

	selected = {
		has_variable = blanquist_cells_prepared
	}

	effect = {
		add_radicals = {
			value = 0.1
		}
		set_variable = blanquist_cells_prepared
		set_variable = {
			name = blanquist_interaction_cooldown
			value = 12
			months = 12
		}
	}

	ai_chance = {
		value = 1
	}
}

	visible = {
		has_journal_entry = je_blanquist_consolidation
		NOT = { has_law = law_type:law_blanquist_bureaucracy }
	}

	possible = {
		NOT = { is_enacting_law = law_type:law_blanquist_bureaucracy }
	}

	selected = {
		OR = {
			has_law = law_type:law_blanquist_bureaucracy
			is_enacting_law = law_type:law_blanquist_bureaucracy
		}
	}

	effect = {

			limit = { enacting_any_law = yes }

		add_enactment_modifier = {
			name = blanquist_bureaucratic_reform_drive
		}
	}

	ai_chance = {
		value = 1
	}
}

je_mutualist_install_petite_bourgeoisie_leader = {
	name = je_mutualist_install_petite_bourgeoisie_leader
	desc = je_mutualist_install_petite_bourgeoisie_leader_desc

	visible = {
		has_journal_entry = je_mutualist_worker_councils
		NOT = { has_variable = mutualist_petite_bourgeoisie_leader_installed }
	}

	possible = {
		ig:ig_petty_bourgeoisie ?= {
			is_marginal = no
			custom_tooltip = {
				text = mutualist_pb_leader_not_conservative_tt
				leader = {
					NOT = {
						OR = {
							has_ideology = ideology:ideology_traditionalist
							has_ideology = ideology:ideology_traditionalist_minoritarian
							has_ideology = ideology:ideology_royalist
							has_ideology = ideology:ideology_authoritarian
							has_ideology = ideology:ideology_fascist
							has_ideology = ideology:ideology_ethno_nationalist
							has_ideology = ideology:ideology_theocrat
							has_ideology = ideology:ideology_integralist
							has_ideology = ideology:ideology_carlist
							has_ideology = ideology:ideology_carlist_2
							has_ideology = ideology:ideology_miguelist
							has_ideology = ideology:ideology_miguelist_2
						}
					}
				}
			}
			leader = {
				NOT = {
					law_stance = {
						law = law_type:law_monarchy
						value > neutral
					}
				}
			}
		}
	}

	selected = {
		has_variable = mutualist_petite_bourgeoisie_leader_installed
	}

	effect = {
		custom_tooltip = {
			text = mutualist_petite_bourgeoisie_leader_appears_tt
			create_character = {
				ig_leader = yes
				interest_group = ig:ig_petty_bourgeoisie
				ideology = ideology_mutualist
				traits = {
					persistent
					meticulous
				}
			}
		}
		set_variable = mutualist_petite_bourgeoisie_leader_installed
		trigger_event = { id = intl_communism.5 popup = yes }
	}

	ai_chance = {
		value = 1
	}
}
