### American Exceptionalism

je_american_exceptionalism_button = {
	name = "je_american_exceptionalism_button"
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					american_exceptionalism_agenda_value >= 1
				}
				desc = je_american_exceptionalism_button_desc_agendas
			}
			triggered_desc = {
				desc = je_american_exceptionalism_button_desc
			}
		}
	}

	visible = {
		NOT = { has_variable = je_american_exceptionalism_declared }
	}

	possible = {
		custom_tooltip = {
			text = je_american_exceptionalism_agendas_tt
			american_exceptionalism_agenda_value >= 5
		}
	}

	effect = {
		set_variable = je_american_exceptionalism_declared
		add_modifier = {
			name = land_of_the_free_home_of_the_brave
			years = 10
			multiplier = american_exceptionalism_agenda_value
		}
	}

	ai_chance = {
		value = 100
	}
}

### Ostend Manifesto

je_ostend_usa_rejects_button = {
	name = "je_ostend_usa_rejects_button"
	desc = "je_ostend_usa_rejects_button_desc"

	visible = {
		c:USA ?= THIS
		is_involved_in_journal_entry = je_ostend_manifesto
		NOT = { has_variable = ostend_usa_rejected }
	}

	possible = {
		hidden_trigger = {
			NOT = { has_variable = ostend_usa_rejected }
			NOT = {
				OR = {
					c:CUB = {
						OR = {
							is_subject_of = c:USA
							NOT = { is_subject_of = c:SPA }
						}
					}
					c:PCO = {
						OR = {
							is_subject_of = c:USA
							NOT = { is_subject_of = c:SPA }
						}
					}
				}
			}
		}
		custom_tooltip = {
			text = ostend_rejection_cooldown_tt
			NOT = { has_variable = ostend_rejection_cooldown }
		}
		custom_tooltip = {
			text = ostend_conclude_blocks_rejection_tt
			NOT = {
				OR = {
					c:CUB = {
						OR = {
							is_subject_of = c:USA
							NOT = { is_subject_of = c:SPA }
						}
					}
					c:PCO = {
						OR = {
							is_subject_of = c:USA
							NOT = { is_subject_of = c:SPA }
						}
					}
				}
			}
		}
	}

	effect = {
		custom_tooltip = ostend_break_negotiations_fail_tt
		set_variable = ostend_usa_rejected
		set_variable = ostend_negotiations_failed
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				"relations(c:SPA)" < relations_threshold:cordial
			}
			add = 100
		}
	}
}

				taking_loans = no

			}

		}
	}
}
