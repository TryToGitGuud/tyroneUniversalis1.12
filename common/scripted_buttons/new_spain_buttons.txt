je_two_spains_coup_button = {
	name = je_two_spains_coup_button
	desc = je_two_spains_coup_button_desc

	visible = {
		var:two_spains_tension_current_var >= 50
		NOT = {
			has_variable = two_spains_coup_var
		}
		has_law = law_type:law_monarchy
		is_revolutionary = no
		is_secessionist = no
	}

	possible = {
		NOR = {
			hidden_trigger = { has_variable = coup_event_active }
			has_journal_entry = je_ip4_coup
			hidden_trigger = { has_variable = glorious_revolution_var }
		}
		any_scope_character = {
			OR = {
				has_role = general
				has_role = admiral
			}
			custom_tooltip = {
				text = ig_sufficiently_upset_tt
				interest_group = {
					ig_approval < 0
				}
				has_variable = monthly_cached_commander_coup_strength
			}
			trigger_if = {
				limit = {
					has_variable = monthly_cached_commander_coup_strength
				}
				custom_tooltip = {
					text = coup_strength_greater_than_coup_resistance_tt
					var:monthly_cached_commander_coup_strength >= root.country_coup_resistance
				}
			}
		}
	}
	selected = {
	}

	effect = {
		ordered_scope_character = {
			limit = {
				OR = {
					has_role = general
					has_role = admiral
				}
				has_variable = monthly_cached_commander_coup_strength
				character_has_ideology_conflicting_with_ruling_dynasty = yes
				var:monthly_cached_commander_coup_strength >= root.country_coup_resistance
				interest_group = {
					ig_approval < 0
				}
			}
			order_by = popularity
			position = 0
			save_scope_as = temp_golpista
		}
		if = {
			limit = {
				exists = scope:temp_golpista
			}
			scope:temp_golpista ?= {
				set_variable = commander_doing_coup_on_player_behalf
			}
			if = {
				limit = {
					exists = scope:temp_golpista
				}
				create_coup_from_character = {
					CHARACTER = scope:temp_golpista
				}
			}
		}
		set_variable = two_spains_coup_var
	}

	ai_chance = {
		value = 0
	}	
}
