je_internationale_debug_socialism = {
	name = je_internationale_debug_socialism
	desc = je_internationale_debug_socialism_desc

	visible = {
		has_journal_entry = je_internationale_debug
	}

	possible = {
		always = yes
	}

	effect = {
		if = {
			limit = {
				NOT = { has_technology_researched = socialism }
			}
			add_technology_researched = socialism
		}
		if = {
			limit = {
				NOT = { has_journal_entry = je_communism_2 }
			}
			add_journal_entry = { type = je_communism_2 }
		}
	}

	ai_chance = {
		value = 0
	}
}
