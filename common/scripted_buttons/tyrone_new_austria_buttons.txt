unify_germany_button = {
	name = unify_germany_button
	desc = unify_germany_button_desc

	visible = {
		NOT = {
			has_variable = unify_germany_button_var
		}
		OR = {
			exists ?= c:PRU
			exists ?= c:NGF
		}
	}

	possible = {
		has_technology_researched = pan-nationalism
		country_rank >= rank_value:great_power
		c:SGF ?= this
	}

	selected = {
	}

	effect = {
		if = {
			limit = {
				exists ?= c:PRU
			}
			create_diplomatic_play = {
				target_state = s:STATE_BRANDENBURG.region_state:PRU
				type = dp_unify_germany
				add_war_goal = {
					holder = root
					target_country = c:PRU
					type = annex_country
				}
			}
		}

		if = {
			limit = {
				exists ?= c:NGF
			}
			create_diplomatic_play = {
				target_state = s:STATE_BRANDENBURG.region_state:NGF
				type = dp_unify_germany
				add_war_goal = {
					holder = root
					target_country = c:NGF
					type = annex_country
				}
			}
		}

		set_variable = unify_germany_button_var
	}
}
