# ========== Form Italy ==========
# At least 12 states in Italy (owned or by subject), must own Romagna, Naples (Campania), Piedmont; have either Lombardy or Rome (Lazio). Changes tag to Italy and fires "Italian Formation" event.
ita_form_italy_button = {
	name = ita_form_italy_button
	desc = ita_form_italy_button_desc

	cooldown = { years = 99 }

	visible = {
		NOT = { c:ITA ?= root }
	}
	ai_chance = {
		value = 50
	}
	possible = {
		custom_tooltip = {
				text = is_nif_tt
				has_variable = NIF_formed
			}
		# At least 12 Italian state regions (north/south Italian homeland) entirely owned by us or a subject
		any_state = {
				OR = {
					state_region = { is_homeland = cu:north_italian }
					state_region = { is_homeland = cu:south_italian }
				}
				OR = {
					owner = root
					owner = { is_subject_of = root }
				}
			count >= 12
		}
		country_or_subject_owns_entire_state_region = STATE_ABRUZZO
		country_or_subject_owns_entire_state_region = STATE_CAMPANIA
		country_or_subject_owns_entire_state_region = STATE_PIEDMONT
		OR = {
			country_or_subject_owns_entire_state_region = STATE_LOMBARDY
			country_or_subject_owns_entire_state_region = STATE_LAZIO
		}
	}

	effect = {
		change_tag = ITA
		trigger_event = { id = italyEventsTyrone.16 popup = yes }
	}
}

# Form Italy (Garibaldi path): visible when the leader of the Garibaldi republic has garibaldi_char_var and the special trait. Effect: kill Southern Garibaldi (hidden), kill true Garibaldi in Sardinia (hidden), enact Presidential Republic in SAR, fire formation event, then follow-up event 17 makes Garibaldi ruler.
ita_form_italy_garibaldi_button = {
	name = ita_form_italy_garibaldi_button
	desc = ita_form_italy_garibaldi_button_desc

	cooldown = { years = 99 }

	visible = {
		NOT = { c:ITA ?= root }
		has_global_variable = garibaldi_republic_established
	}

	possible = {
		custom_tooltip = {
			text = is_nif_tt
			has_variable = NIF_formed
		}
		any_state = {
				OR = {
					state_region = { is_homeland = cu:north_italian }
					state_region = { is_homeland = cu:south_italian }
				}
				OR = {
					owner = root
					owner = { is_subject_of = root }
				}
			count >= 12
		}
		country_or_subject_owns_entire_state_region = STATE_ABRUZZO
		country_or_subject_owns_entire_state_region = STATE_CAMPANIA
		country_or_subject_owns_entire_state_region = STATE_PIEDMONT
		OR = {
			country_or_subject_owns_entire_state_region = STATE_LOMBARDY
			country_or_subject_owns_entire_state_region = STATE_LAZIO
		}
		custom_tooltip = {
			text = ita_form_garibaldi_tension_tt
			var:italian_sentiment_var >= 90
		}

	}

	effect = {
		random_country = {
			limit = {
				has_variable = garibaldi_exp_country
				ruler = {
					has_variable = garibaldi_char_var
					has_trait = character_trait_garibaldi_republican
				}
			}
			save_scope_as = garibaldi_republic
		}
		scope:garibaldi_republic = {
			ruler = {
				kill_character = { hidden = yes }
			}
		}
		# Kill the true Garibaldi (the general in Sardinia from the expedition)
		random_character = {
			limit = { has_variable = garibaldi_char_var }
			kill_character = { hidden = yes }
		}
		c:SAR = {
			change_tag = ITA
			activate_law = law_type:law_presidential_republic
			trigger_event = { id = italyEventsTyrone.16 popup = yes }
			trigger_event = { id = italyEventsTyrone.17 days = 1 popup = yes }
		}
	}
}
