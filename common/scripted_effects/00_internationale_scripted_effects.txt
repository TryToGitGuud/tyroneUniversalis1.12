try_claim_internationale = {
	if = {
		limit = {
			NOT = { has_global_variable = intl_internationale_claimed }
		}
		set_global_variable = intl_internationale_claimed
		set_global_variable = {
			name = intl_internationale_leader_country
			value = root
		}
		every_country = {
			limit = {
				this = global_var:intl_internationale_leader_country
			}
			set_variable = intl_internationale_leader
		}
		set_global_variable = intl_internationale_$IDEOLOGY$
		if = {
			limit = { has_global_variable = intl_internationale_anarchist }
			intl_unlock_anarchist_laws = yes
		}
		if = {
			limit = { has_global_variable = intl_internationale_mutualist }
			intl_unlock_mutualist_laws = yes
		}
		if = {
			limit = { has_global_variable = intl_internationale_syndicalist }
			intl_unlock_syndicalist_laws = yes
		}
		if = {
			limit = { has_global_variable = intl_internationale_blanquist }
			intl_unlock_blanquist_laws = yes
		}
		if = {
			limit = { has_global_variable = intl_internationale_marxist }
			intl_unlock_marxist_laws = yes
		}
		every_country = {
			limit = {
				NOT = { is_country_type = decentralized }
			}
			trigger_event = { id = intl_communism.23 popup = yes }
		}
	}
}

		target = { save_scope_as = intl_proletarian_war_target }

				any_scope_initiator_ally = {

					count >= 2
					country_rank = rank_value:great_power

# Internationale Conference â€” Law State Effects
# $LAW$ = law key without the "law_" prefix
# e.g. intl_add_law_to_conference = { LAW = blanquist_economy }

# Set law to conference candidate (value = 1)
# No-op if already permanently enabled (value = 2) â€” permanent laws cannot be demoted
intl_add_law_to_conference = {
	if = {
		limit = {
			NOT = {
				has_global_variable = intl_law_$LAW$
				global_var:intl_law_$LAW$ >= 2
			}
		}
		set_global_variable = { name = intl_law_$LAW$ value = 1 }
	}
}

# Remove law from conference (clears the global variable)
# No-op if permanently enabled â€” permanent laws cannot be removed
intl_remove_law_from_conference = {
	if = {
		limit = {
			has_global_variable = intl_law_$LAW$
			global_var:intl_law_$LAW$ < 2
		}
		remove_global_variable = intl_law_$LAW$
	}
}

intl_update_parliament = {

	set_variable = { name = intl_allocated_seats value = 0 }
	set_variable = { name = intl_total_counted_seats value = 0 }
	set_variable = { name = intl_winning_faction_seats value = 0 }
	if = {
		limit = { has_variable = intl_winning_faction }
		remove_variable = intl_winning_faction
	}
	if = {
		limit = { has_variable = intl_leftover_seats }
		remove_variable = intl_leftover_seats
	}

	intl_clear_faction_cache = { FACTION = anarchist }
	intl_clear_faction_cache = { FACTION = mutualist }
	intl_clear_faction_cache = { FACTION = syndicalist }
	intl_clear_faction_cache = { FACTION = blanquist }
	intl_clear_faction_cache = { FACTION = marxist }

	if = { limit = { NOT = { has_variable = intl_strength_anarchist } } set_variable = { name = intl_strength_anarchist value = 0 } }
	if = { limit = { NOT = { has_variable = intl_strength_mutualist } } set_variable = { name = intl_strength_mutualist value = 0 } }
	if = { limit = { NOT = { has_variable = intl_strength_syndicalist } } set_variable = { name = intl_strength_syndicalist value = 0 } }
	if = { limit = { NOT = { has_variable = intl_strength_blanquist } } set_variable = { name = intl_strength_blanquist value = 0 } }
	if = { limit = { NOT = { has_variable = intl_strength_marxist } } set_variable = { name = intl_strength_marxist value = 0 } }

	set_variable = {
		name = intl_total_strength
		value = {
			value = var:intl_strength_anarchist
			add = var:intl_strength_mutualist
			add = var:intl_strength_syndicalist
			add = var:intl_strength_blanquist
			add = var:intl_strength_marxist
		}
	}

	if = {
		limit = {
			var:intl_total_strength > 0
		}

		intl_cache_faction_seats = { FACTION = anarchist }
		intl_cache_faction_seats = { FACTION = mutualist }
		intl_cache_faction_seats = { FACTION = syndicalist }
		intl_cache_faction_seats = { FACTION = blanquist }
		intl_cache_faction_seats = { FACTION = marxist }

		set_variable = {
			name = intl_leftover_seats
			value = {
				value = intl_number_of_seats
				subtract = var:intl_total_counted_seats
				min = 0
			}
		}

		intl_allocate_remaining = { FACTION = anarchist }
		intl_allocate_remaining = { FACTION = mutualist }
		intl_allocate_remaining = { FACTION = syndicalist }
		intl_allocate_remaining = { FACTION = blanquist }
		intl_allocate_remaining = { FACTION = marxist }

		set_variable = { name = intl_allocated_seats value = 0 }

		intl_apply_faction_range = { FACTION = anarchist }
		intl_apply_faction_range = { FACTION = mutualist }
		intl_apply_faction_range = { FACTION = syndicalist }
		intl_apply_faction_range = { FACTION = blanquist }
		intl_apply_faction_range = { FACTION = marxist }
	}
}

intl_clear_faction_cache = {
	set_variable = { name = intl_seat_count_$FACTION$ value = 0 }
	set_variable = { name = intl_$FACTION$_parl_min value = 0 }
	set_variable = { name = intl_$FACTION$_parl_max value = -1 }
}

intl_cache_faction_seats = {
	set_variable = {
		name = intl_seat_count_$FACTION$
		value = {
			value = var:intl_strength_$FACTION$
			divide = var:intl_total_strength
			multiply = intl_number_of_seats
			floor = yes
		}
	}
	change_variable = {
		name = intl_total_counted_seats
		add = var:intl_seat_count_$FACTION$
	}
	if = {
		limit = {
			var:intl_seat_count_$FACTION$ > var:intl_winning_faction_seats
		}
		set_variable = { name = intl_winning_faction_seats value = var:intl_seat_count_$FACTION$ }
		set_variable = { name = intl_winning_faction value = flag:$FACTION$ }
	}
}

intl_allocate_remaining = {
	if = {
		limit = {
			has_variable = intl_winning_faction
			var:intl_leftover_seats > 0
			var:intl_winning_faction = flag:$FACTION$
		}
		change_variable = {
			name = intl_seat_count_$FACTION$
			add = var:intl_leftover_seats
		}
	}
}

intl_apply_faction_range = {
	if = {
		limit = {
			var:intl_seat_count_$FACTION$ > 0
		}
		set_variable = {
			name = intl_$FACTION$_parl_min
			value = var:intl_allocated_seats
		}
		set_variable = {
			name = intl_$FACTION$_parl_max
			value = {
				value = var:intl_allocated_seats
				add = var:intl_seat_count_$FACTION$
				subtract = 1
			}
		}
		change_variable = {
			name = intl_allocated_seats
			add = var:intl_seat_count_$FACTION$
		}
	}
}

## script_parameter = SIDE
intl_cast_vote = {
	# Remove from whichever list this country was previously on
	if = {
		limit = { has_variable = intl_vote_stance  var:intl_vote_stance = 1 }
		remove_list_global_variable = { name = intl_vote_keep_list target = this }
	}
	if = {
		limit = { has_variable = intl_vote_stance  var:intl_vote_stance = 2 }
		remove_list_global_variable = { name = intl_vote_proposed_list target = this }
	}
	set_variable = { name = intl_vote_stance value = $SIDE$ }
	# Add to new list
	if = {
		limit = { var:intl_vote_stance = 1 }
		add_to_global_variable_list = { name = intl_vote_keep_list target = this }
	}
	if = {
		limit = { var:intl_vote_stance = 2 }
		add_to_global_variable_list = { name = intl_vote_proposed_list target = this }
	}
	intl_update_vote_prestige = yes
}

intl_update_vote_prestige = {
	set_global_variable = { name = intl_vote_prestige_current value = 0 }
	set_global_variable = { name = intl_vote_prestige_proposed value = 0 }
	set_global_variable = { name = intl_vote_count_current value = 0 }
	set_global_variable = { name = intl_vote_count_proposed value = 0 }
	every_country = {
		limit = {
			OR = {
				has_journal_entry = je_internationale
				has_journal_entry = je_mp_faction_world
			}
			has_variable = intl_vote_stance
		}
		if = {
			limit = { intl_country_supports_current = yes }
			change_global_variable = {
				name = intl_vote_prestige_current
				add = { value = prestige }
			}
			change_global_variable = { name = intl_vote_count_current add = 1 }
		}
		if = {
			limit = { intl_country_supports_proposed = yes }
			change_global_variable = {
				name = intl_vote_prestige_proposed
				add = { value = prestige }
			}
			change_global_variable = { name = intl_vote_count_proposed add = 1 }
		}
	}
	# Derived variables for the double-direction progress bar
	# left side needs a negative value; total needed for min/max bounds
	set_global_variable = { name = intl_vote_prestige_current_neg value = 0 }
	change_global_variable = {
		name = intl_vote_prestige_current_neg
		add = { value = global_var:intl_vote_prestige_current multiply = -1 }
	}
	set_global_variable = {
		name = intl_vote_prestige_total
		value = { value = global_var:intl_vote_prestige_current add = global_var:intl_vote_prestige_proposed }
	}
	set_global_variable = {
		name = intl_vote_prestige_total_neg
		value = { value = global_var:intl_vote_prestige_total multiply = -1 }
	}
}
