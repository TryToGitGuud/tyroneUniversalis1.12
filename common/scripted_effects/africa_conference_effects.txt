get_ahead_africa_conference = {
	set_variable = is_ahead_berlin_conf
	random_list = {
			50 = {
				trigger = {
					NOT = {
						has_variable = treaty_blocked_niger_africa
						any_country = { 
							has_variable = claimed_niger_var 
						} 
					}
				}
				set_variable = claimed_niger_var
				add_to_global_variable_list = { name = niger_claimed_countries_list target = this }
				change_variable = { name = remaining_claims_var subtract = 1 }
				change_variable = { name = berlin_claimed_count add = 1 }
			}
			50 = {
				trigger = {
					NOT = {
						has_variable = treaty_blocked_south_africa
						any_country = { 
							has_variable = claimed_south_africa_var 
						} 
					}
				}
				set_variable = claimed_south_africa_var

			}
