ita_industry_calculate_sluggish_effect = {
	change_variable = {
			name = agri_padana_ita_var
			add = var:investment_in_padana_ita_var
		}
	if = {
		limit = { 
			var:investment_in_north_ita_var = var:investment_in_south_ita_var 
			var:investment_in_north_ita_var > 0
			var:investment_in_south_ita_var > 0
		}
		change_variable = {
			name = industrious_north_ita_var
			add = 1
			}
		change_variable = {
			name = sluggish_south_ita_var
			add = 1
			}
	}
	else = {
		change_variable = {
			name = industrious_north_ita_var
			add = var:investment_in_north_ita_var
			}
		change_variable = {
			name = industrious_north_ita_var
			subtract = var:investment_in_south_ita_var
			}

		change_variable = {
			name = sluggish_south_ita_var
			add = var:investment_in_south_ita_var
			}
		change_variable = {
			name = sluggish_south_ita_var
			subtract = var:investment_in_north_ita_var
			}
		}
		
	

	clamp_variable = { name = sluggish_south_ita_var max = 20 min = -20 }
	clamp_variable = { name = industrious_north_ita_var max = 20 min = -20 }
	clamp_variable = { name = agri_padana_ita_var max = 50 min = 0 }

	set_variable = {
		name = investment_in_north_ita_var
		value = 0
	}
	set_variable = {
		name = investment_in_padana_ita_var
		value = 0
	}
	set_variable = {
		name = investment_in_south_ita_var
		value = 0
	}

}
