# Berlin Conference - Block Claiming treaties
# source = the proposer (has leverage); target = the blocked country
# on_entry_into_force sets treaty_blocked_X on target
# on_withdrawal removes it, regardless of who withdraws

berlin_block_west_africa = {
	kind = directed
	cost = 25
	maintenance_paid_by = source_country
	relations_progress_per_day = 0.5

	icon = "gfx/interface/icons/diplomatic_action_icons/doctrine_of_lapse.dds"

	unlocked_by_technologies = { colonization }

	visible = {
		root = {
			is_involved_in_journal_entry = je_berlin_conference
			has_variable = visible_berlin_conf_treaty_article
		}
	}

	possible = {
		root = { can_claim_west_africa = yes }
	}

	can_ratify = {
		NOT = { has_global_variable = berlin_conference_ended }
	}

	on_entry_into_force = {
		custom_tooltip = { text = berlin_block_removes_claim_tt }
		scope:article_options = {
			target_country = { set_variable = treaty_blocked_west_africa }
			source_country = {
				if = {
					limit = { has_variable = claimed_west_africa_var }
					remove_variable = claimed_west_africa_var
					remove_list_global_variable = { name = west_africa_claimed_countries_list target = this }
					change_variable = { name = remaining_claims_var add = 1 }
				}
			}
		}
	}

	on_withdrawal = {
		if = {
			limit = { scope:non_withdrawing_country = { has_variable = treaty_blocked_west_africa } }
			scope:non_withdrawing_country = { remove_variable = treaty_blocked_west_africa }
		}
		else_if = {
			limit = { scope:withdrawing_country = { has_variable = treaty_blocked_west_africa } }
			scope:withdrawing_country = { remove_variable = treaty_blocked_west_africa }
		}
	}

	ai = {
		treaty_categories = { expansion }
		evaluation_chance = { value = 0.5 }

		proposal_weight = {
			value = 0
			if = {
				limit = { has_variable = claimed_west_africa_var }
				add = 2
			}
		}

		article_ai_usage = { offer }

		inherent_accept_score = {
			value = 0

			save_temporary_scope_as = target
			scope:other_country = {
				save_temporary_scope_as = actor
			}

			add = {
				desc = "berlin_accept_base"
				value = -50
			}

			if = {
				limit = { scope:target_country ?= this }
				add = {
					desc = "berlin_accept_wants_to_offer"
					value = 40
				}

				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_ATTITUDE"
					if = {
						limit = {
							has_attitude = {
								who = scope:other_country
								attitude = protective
							}
						}
						value = -30
					}
					else_if = {
						limit = {
							OR = {
								has_attitude = {
									who = scope:other_country
									attitude = antagonistic
								}
								has_attitude = {
									who = scope:other_country
									attitude = belligerent
								}
								has_attitude = {
									who = scope:other_country
									attitude = domineering
								}
							}
						}
						value = 40
					}
					else_if = {
						limit = {
							OR = {
								has_attitude = {
									who = scope:other_country
									attitude = wary
								}
								has_attitude = {
									who = scope:other_country
									attitude = disinterested
								}
							}
						}
						value = 20
					}
				}
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_ATTITUDE"
				if = {
					limit = {
						has_attitude = {
							who = scope:other_country
							attitude = protective
						}
					}
					value = 25
				}
				else_if = {
					limit = {
						OR = {
							has_attitude = {
								who = scope:other_country
								attitude = antagonistic
							}
							has_attitude = {
								who = scope:other_country
								attitude = belligerent
							}
							has_attitude = {
								who = scope:other_country
								attitude = domineering
							}
						}
					}
					value = -100
				}
				else_if = {
					limit = {
						OR = {
							has_attitude = {
								who = scope:other_country
								attitude = wary
							}
							has_attitude = {
								who = scope:other_country
								attitude = disinterested
							}
						}
					}
					value = -50
				}
			}

			if = {
				limit = { has_variable = claimed_west_africa_var }
				add = {
					desc = "berlin_accept_has_claim"
					value = -30
				}
			}
			else = {
				add = {
					desc = "berlin_accept_no_claim"
					value = 25
				}
			}

			if = {
				limit = { country_rank >= rank_value:great_power }
				add = {
					desc = "berlin_accept_gp_rank"
					value = -50
				}
			}

			if = {
				limit = { has_colonial_growth = yes }
				add = {
					desc = "berlin_accept_colonial_ambition"
					value = -40
				}
			}

			if = {
				limit = { has_strategy = ai_strategy_colonial_expansion }
				add = {
					desc = "berlin_accept_colonial_strategy"
					value = -25
				}
			}

			if = {
				limit = {
					scope:source_country = {
						is_country_type = unrecognized
					}
				}
				add = {
					desc = "concept_unrecognized_power_short"
					value = -50
				}
			}

		}
	}
}
