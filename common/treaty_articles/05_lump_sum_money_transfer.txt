lump_sum_money_transfer = {
	kind = directed
	cost = 25

	icon = "gfx/interface/icons/diplomatic_treaties_articles_icons/bankroll_treaties.dds"

	flags = {
		can_be_enforced
		giftable
		ai_consider_threaten_naval_hostilities
	}

	maintenance_paid_by = source_country
	usage_limit = once_per_treaty

	required_inputs = {
		quantity
	}

	unlocked_by_technologies = {
		central_banking
	}

	visible = {
	}

	possible = {
		in_default = no
	}

	requirement_to_maintain = {
		trigger = {
			custom_tooltip = {
				text = "LUMP_SUM_MONEY_TRANSFER_INPUT_QUANTITY_INVALID"
				scope:article ?= {
					exists = input_quantity
					input_quantity >= 1000
				}
			}
		}
	}

	non_fulfillment = {}

	can_ratify = {
		NOT = {
			scope:treaty ?= {
				any_scope_article_option = {
					has_type = take_on_debt
					hidden_trigger = {
						source_country = scope:target_country
					}
				}
			}
		}
		scope:source_country = {
			in_default = no
		}
		custom_tooltip = {
			text = CANT_SEND_MONEY_TOO_LITTLE_CASH_ON_HAND
			scope:source_country = {
				imperia_current_transferable_money_value >= scope:article.input_quantity
			}
		}
		custom_tooltip = {
			text = CANT_SEND_MONEY_TOO_LITTLE_SPACE_IN_TARGET
			scope:target_country = {
				imperia_current_max_of_receivable_money_value >= scope:article.input_quantity
			}
		}
	}

	on_entry_into_force = {
		scope:article_options = {
			source_country = {
				add_treasury = {
					value = scope:article_options.input_quantity
					multiply = -1
				}
			}
			target_country = {
				add_treasury = scope:article_options.input_quantity
			}
		}
	}

	on_withdrawal = {}
	on_break = {
		scope:withdrawing_country = {
			change_infamy = {
				value = {
					add = 1
					multiply = {
						value = scope:treaty_options.remaining_binding_period
						min = 365
						divide = 365
					}
					max = 20
					min = 5
				}
			}
		}
		scope:withdrawing_country = {
			change_relations = { country = scope:non_withdrawing_country value = -10 }
		}
	}

	quantity_min_value = {
		value = 1000
	}

	quantity_max_value = {
		value = 2000000
		max = imperia_current_transferable_money_value
		min = 1000
	}

	ai = {
		treaty_categories = {
			none
		}

		quantity_input_value = {
			value = 2000000
			max = imperia_current_transferable_money_value
			min = 1000
		}

		evaluation_chance = {
			value = 0.0
		}

		article_ai_usage = {
			request
		}

		inherent_accept_score = {
			value = -1000

			if = {
				limit = {
					scope:source_country ?= this
					exists = scope:article.input_quantity
				}
				subtract = {
					desc = "DIPLOMATIC_ACCEPTANCE_INPUT_QUANTITY"
					value = scope:article.input_quantity
					divide = 20000
				}
			}
			else_if = {
				limit = {
					exists = scope:article.input_quantity
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_INPUT_QUANTITY"
					value = scope:article.input_quantity
					divide = 20000
					max = 100
				}
			}
		}
	}

	wargoal = {
		execution_priority = 21

		contestion_type = control_any_target_incorporated_state

		maneuvers = {
			value = 0

			add = {
				desc = "MANEUVERS_BASE_VALUE"
				value = 2.5
			}

			if = {
				limit = {
					exists = scope:quantity
				}
				add = {
					desc = "MANEUVERS_QUANTITY_FACTOR"
					value = scope:quantity
					divide = 1000
					min = 0
					max = 50
				}
			}
		}

		infamy = {
			value = 0

			if = {
				limit = {
					exists = scope:quantity
				}
				add = {
					desc = "INFAMY_QUANTITY_FACTOR"
					value = scope:quantity
					divide = 2000
					min = 5
					max = 50
				}
			}
		}
	}
}
