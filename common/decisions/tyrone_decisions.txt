matter_of_hungary_federate_three_crowns_decision = {

	is_shown = {
		NOT = {
			has_variable = federated_three_crowns_var
		}
		has_technology_researched = nationalism
		NOT = {
			has_journal_entry = je_matter_of_hungary
		}
		c:AUS ?= this
		c:HUN ?= {
			is_subject_of = root
		}
		c:CRO ?= {
			is_subject_of = root
		}
	}

	possible = {
		custom_tooltip = {
			text = matter_of_hungary_three_crowns_button_tt
			AND = {
				any_subject_or_below = {
					c:HUN ?= this
					OR = {
						liberty_desire < 25
						relations:root >= relations_threshold:amicable	
					}
				}
				any_subject_or_below = {
					c:CRO ?= this
					OR = {
						liberty_desire < 25
						relations:root >= relations_threshold:amicable	
					}
				
				}
			}
		}
		has_government_type = gov_triple_monarchy
		is_at_war = no
		is_diplomatic_play_committed_participant = no
	}

		c:HUN = {
			save_temporary_scope_as = hungary_scope
		}
		c:CRO = {
			save_temporary_scope_as = croatia_scope
		}
		set_variable = federated_three_crowns_var
		every_subject_or_below = {
			limit = {
				OR = {
					is_subject_type = subject_type_personal_union
					is_subject_type = subject_type_crown_land
					is_subject_type = subject_type_puppet
				}
				capital = {
					OR = {
						region = sr:region_danubia
						region = sr:region_balkans
					}
				}
			}
			c:AUS ?= {
				annex_with_incorporation = prev
			}
		}
		if = {
			limit = {
				has_law_or_variant = law_type:law_monarchy
			}
			custom_tooltip = {
				text = change_tag_to_KUK_triple_monarchy_tt
			}
		}
		danubian_federation_integrate_culture = {
			CULTURE = cu:hungarian
		}
		danubian_federation_integrate_culture = {
			CULTURE = cu:croat
		}
		add_loyalists = {
			culture = cu:croat
			value = medium_radicals
		}
		add_loyalists = {
			culture = cu:hungarian
			value = medium_radicals
		}
		add_radicals = {
			culture = cu:romanian
			value = medium_radicals
		}
	}

	ai_chance = {
		value = 10
		if = {
			limit = { # Will not annex players until their liberty desire is below 5.
				any_subject_or_below = {
					is_subject_type = subject_type_crown_land
					is_player = yes
					liberty_desire > 5
				}
			}
			add = -100
		}
	}
}
