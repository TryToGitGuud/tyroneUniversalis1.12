claim_south_africa_subject = {
	is_shown = {
		is_ai = yes
		is_involved_in_journal_entry = je_berlin_conference
		NOT = { has_global_variable = berlin_conference_ended }
		NOT = { has_variable = claimed_south_africa_var }
		NOT = { has_variable = treaty_blocked_south_africa }
		NOT = { any_country = { has_variable = claimed_south_africa_var } }
		any_subject_or_below = {
			capital = {
                is_in_geographic_region = geographic_region_south_africa
            }
		}
		
	}

	possible = {
		
	}

	when_taken = {
		set_variable = claimed_south_africa_var
		add_to_global_variable_list = { name = south_africa_claimed_countries_list target = this }
		change_variable = { name = berlin_claimed_count add = 1 }
	}

	ai_chance = {
		value = 100
	}
}

	}

	possible = {
		
	}

	when_taken = {
		random_list = {
			90 = {
			}
			3 = {

					text = berlin_conf_accept

	}

	}
}

berlin_getahead1_conference = {
	is_shown = {
		is_ai = yes
		is_involved_in_journal_entry = je_berlin_conference
		OR = {
			var:conference_gp_number = 1
			AND = {
				var:conference_gp_number = 2
				any_country = { 
					has_variable = is_ahead_berlin_conf 
					var:conference_gp_number = 1
				} 
			}
		}
		NOT = { has_variable = is_ahead_berlin_conf }
	}

	possible = {
		
	}

	when_taken = {
		set_variable = is_ahead_berlin_conf
		random_list = {
			50 = {

				change_variable = { name = berlin_claimed_count add = 1 }
			}
			50 = {
				trigger = {
					NOT = {
						has_variable = treaty_blocked_south_africa
						any_country = { 
							has_variable = claimed_south_africa_var 
						} 
					}
				}
				set_variable = claimed_south_africa_var
				add_to_global_variable_list = { name = south_africa_claimed_countries_list target = this }
				change_variable = { name = remaining_claims_var subtract = 1 }
				change_variable = { name = berlin_claimed_count add = 1 }
			}
			50 = {
				trigger = {
					NOT = {
						has_variable = treaty_blocked_kenya_africa
						any_country = {
