# $LAW$ = law key without the "law_" prefix
# e.g. intl_law_in_conference = { LAW = blanquist_economy }

# --- 00_coa_triggers.txt ---
# Override vanilla communist flag trigger so Internationale council variants count for communist COAs.
coa_communist_trigger = {
	OR = {
		has_law = law_type:law_council_republic
		has_law = law_type:law_anarchist_council_republic
		has_law = law_type:law_marxist_council_republic
		has_law = law_type:law_mutualist_council_republic
		has_law = law_type:law_syndicalist_council_republic
		has_law = law_type:law_blanquist_presidential_republic
	}
}

# True when a law is a live conference candidate (value = 1, not yet finalized)
## script_parameter = LAW
intl_law_in_conference = {
	has_global_variable = intl_law_$LAW$
	global_var:intl_law_$LAW$ >= 1
	global_var:intl_law_$LAW$ < 2
}

# True when a law has been permanently enabled by the conference (value >= 2)
# This is the gate used in each law's can_enact block
## script_parameter = LAW
intl_law_permanently_enabled = {
	has_global_variable = intl_law_$LAW$
	global_var:intl_law_$LAW$ >= 2
}

# --- intl_vote_triggers.txt ---
# Internationale Conference â€” Vote State Triggers

## script_parameter = CURRENT
## script_parameter = PROPOSED
intl_vote_can_start = {
	NOT = { has_global_variable = intl_vote_active }
	intl_law_permanently_enabled = { LAW = $CURRENT$ }
}

intl_vote_is_active = {
	has_global_variable = intl_vote_active
}

intl_country_has_voted = {
	has_variable = intl_vote_stance
}

intl_country_supports_current = {
	has_variable = intl_vote_stance
	var:intl_vote_stance = 1
}

intl_country_supports_proposed = {
	has_variable = intl_vote_stance
	var:intl_vote_stance = 2
}

intl_is_internationale_leader = {
	has_global_variable = intl_internationale_leader_country
	this = global_var:intl_internationale_leader_country
}

intl_has_current_internationale_line = {
	OR = {
		AND = {
			has_global_variable = intl_internationale_anarchist
			has_law = law_type:law_anarchist_council_republic
		}
		AND = {
			has_global_variable = intl_internationale_mutualist
			has_law = law_type:law_mutualist_council_republic
		}
		AND = {
			has_global_variable = intl_internationale_syndicalist
			has_law = law_type:law_syndicalist_council_republic
		}
		AND = {
			has_global_variable = intl_internationale_blanquist
			has_law = law_type:law_blanquist_presidential_republic
		}
		AND = {
			has_global_variable = intl_internationale_marxist
			has_law = law_type:law_marxist_council_republic
		}
	}
}

intl_can_receive_spread_the_revolution = {
	NOT = { is_country_type = decentralized }
	NOT = { intl_has_current_internationale_line = yes }
}

intl_is_great_proletarian_war_candidate = {
	country_rank = rank_value:great_power
	intl_has_current_internationale_line = yes
	is_at_war = yes
	any_scope_war = {
		any_war_participant = {
			count >= 2
			country_rank = rank_value:great_power
			intl_has_current_internationale_line = yes
		}
		any_war_participant = {
			count >= 2
			country_rank = rank_value:great_power
			NOT = { intl_has_current_internationale_line = yes }
		}
	}
}

intl_great_proletarian_war_can_escalate = {
	any_country = {
		intl_is_great_proletarian_war_candidate = yes
	}
}

intl_diplomatic_play_has_great_proletarian_war_shape = {
	OR = {
		AND = {
			any_scope_initiator_ally = {
				count >= 2
				country_rank = rank_value:great_power
				has_law_or_variant = law_type:law_council_republic
			}
			any_scope_target_ally = {
				count >= 2
				country_rank = rank_value:great_power
				NOT = { has_law_or_variant = law_type:law_council_republic }
			}
		}
		AND = {
			any_scope_target_ally = {
				count >= 2
				country_rank = rank_value:great_power
				has_law_or_variant = law_type:law_council_republic
			}
			any_scope_initiator_ally = {
				count >= 2
				country_rank = rank_value:great_power
				NOT = { has_law_or_variant = law_type:law_council_republic }
			}
		}
	}
}
