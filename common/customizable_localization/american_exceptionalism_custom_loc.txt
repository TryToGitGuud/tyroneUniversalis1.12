	type = country
	text = {
		trigger = {
			NOT = {
				has_variable = je_manifest_destiny_mexico_complete
				has_variable = je_manifest_destiny_frontier_wars_complete
			}
		}
		localization_key = metternich_unticked
	}
	text = {
		trigger = {
			has_variable = je_manifest_destiny_mexico_complete
			has_variable = je_manifest_destiny_frontier_wars_complete
		}
		localization_key = metternich_ticked
	}
}
