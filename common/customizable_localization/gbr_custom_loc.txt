get_commonwealth_can_progress = {
	type = journal_entry
	text = {
		trigger = { "scripted_bar_progress(bar_can_commonwealth)" >= 300 }
		localization_key = commonwealth_bar_stage3_key
	}
	text = {
		trigger = { "scripted_bar_progress(bar_can_commonwealth)" >= 200 }
		localization_key = commonwealth_bar_stage2_key
	}
	text = {
		trigger = { "scripted_bar_progress(bar_can_commonwealth)" >= 100 }
		localization_key = commonwealth_bar_stage1_key
	}
	text = {
		localization_key = commonwealth_bar_stage0_key
	}
}
