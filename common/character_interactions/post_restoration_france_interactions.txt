enact_law_enforce_napoleon = {
	icon = "gfx/interface/icons/event_icons/napoleon_empire_icon.dds"

	potential = {
		is_ruler = yes
		owner ?= scope:actor
		owner ?= c:FRA
		age >= define:NCharacters|ADULT_AGE
		owner ?= { has_journal_entry = je_post_bonapartists }
	}

	possible = {
		owner ?= {
			enacting_any_law = yes
			authority >= 200
		}
		custom_tooltip = {
			text = last_five_years_not_ENFORCED
			NOT = {
				has_variable = last_five_years_ENFORCED
			}
		}
		trigger_if = {
			limit = {
				owner ?= {
					enacting_any_law = yes
				}
			}
			owner ?= {
				enactment_phase >= 1
			}
			OR = {
				law_enactment_stance = {
					law = owner.currently_enacting_law.type
					value >= neutral
				}
				owner ?= {
					any_political_movement = {
						movement_supports_enaction = yes
						is_revolutionary = yes
					}
				}
			}
		}
	}

	effect = {
		owner = {
			activate_law = owner.currently_enacting_law.type
		}
		add_modifier = {
			name = napoleon_decree_law
			years = 5
			is_decaying = yes
		}
		set_variable = {
			name = last_five_years_ENFORCED
			years = 5
		}
	}
}
