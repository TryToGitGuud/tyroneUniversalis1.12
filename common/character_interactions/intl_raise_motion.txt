intl_raise_motion_anarchist = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	potential = {
		has_variable = bakunin_var
		owner ?= {
			OR = {
				has_journal_entry = je_internationale
				has_journal_entry = je_mp_faction_world
			}
		}
	}

	possible = {
		scope:actor = {
			custom_tooltip = {
				text = intl_raise_motion_no_active_vote_tt
				NOT = { intl_vote_is_active = yes }
			}
			custom_tooltip = {
				text = intl_raise_motion_no_cooldown_tt
				NOT = { has_variable = intl_motion_cooldown_months }
			}
		}
		custom_tooltip = {
			text = intl_raise_motion_anarchist_laws_tt
			OR = {
				AND = {
					intl_law_permanently_enabled = { LAW = marxist_council_republic }
					NOT = { intl_law_permanently_enabled = { LAW = anarchist_council_republic } }
				}
				AND = {
					intl_law_permanently_enabled = { LAW = marxist_economy }
					NOT = { intl_law_permanently_enabled = { LAW = anarchist_economy } }
				}
				AND = {
					intl_law_permanently_enabled = { LAW = mutualist_economy }
					NOT = { intl_law_permanently_enabled = { LAW = anarchist_economy } }
				}
			}
		}
	}

	effect = {
		scope:actor = {
			trigger_event = { id = intl_vote_events.11 popup = yes }
		}
	}

	ai_chance = {
		value = 0
	}
}
