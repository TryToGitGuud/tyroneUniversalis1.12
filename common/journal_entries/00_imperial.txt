je_imperial_federation = {
	icon = "gfx/interface/icons/event_icons/imperial_federation.dds"

	group = je_group_global_international_situations

	should_be_involved = {
		is_in_british_empire = yes
	}

	should_show_when_not_involved = {
		OR = {
			capital = {
				region = sr:region_england
			}
            top_overlord ?= {
                capital = {
					region = sr:region_england
                }
            }
		}
	}

	is_shown_in_lobby = {
		c:GBR = THIS
	}

	is_shown_when_inactive = {
		NOT = {
			has_global_variable = imp_fed_started
			c:GBR ?= THIS
		}
	}

	possible = {
		c:GBR ?= {
			has_technology_researched = pan-nationalism
			has_technology_researched = civilizing_mission
			custom_tooltip = {
				text = federated_canada_tt
				has_variable = canada_federated_var
			}
			custom_tooltip = {
				text = federated_australia_tt
				has_variable = australia_federated_var
			}
		}
		c:CAN ?= {
			exists = yes
			is_subject_of = c:GBR
			OR = {
				is_subject_type = subject_type_colony
				is_subject_type = subject_type_dominion
			}
		}
		c:AST ?= {
			exists = yes
			is_subject_of = c:GBR
			OR = {
				is_subject_type = subject_type_colony
				is_subject_type = subject_type_dominion
			}
		}
		c:SAF ?= {
			exists = yes
			is_subject_of = c:GBR
			OR = {
				is_subject_type = subject_type_colony
				is_subject_type = subject_type_dominion
			}
		}
		c:BIC ?= {
			exists = yes
			is_subject_of = c:GBR
			OR = {
				is_subject_type = subject_type_colony
				is_subject_type = subject_type_dominion
			}
		}
	}


	timeout = 7300

	#scripted_button = approach_canadian_officials
	#scripted_button = approach_saf_officials
	#scripted_button = approach_australian_officials
	scripted_button = enact_imperial_federation

	immediate = {
		imperial_federation_calculate_shared_laws = yes
		c:GBR = {
			save_scope_as = gbr_scope
			trigger_event = {
				id = imperial_events.1
			}
		}
		set_global_variable = imp_fed_started
	}

	on_monthly_pulse = {
		effect = {
			imperial_federation_calculate_shared_laws = yes
		}
		random_events = {
			#TODO:
		}
	}

	complete = {
		custom_tooltip = {
			text = imperial_federation_formed_tt
			has_global_variable = imperial_federation_var
		}
	}

	on_complete = {
		trigger_event = {
			id = imperial_events.2 #india gets angry but imperial fed is cool
		}
	}

	on_timeout = {
		trigger_event = {
			id = imperial_events.3 #failure
		}
	}

	invalid = {
		has_global_variable = imp_fed_cancelled
	}

	weight = 500

	should_be_pinned_by_default = yes
}
