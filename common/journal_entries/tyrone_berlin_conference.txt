je_berlin_conference = {
	icon = "gfx/interface/icons/event_icons/berlin_conference.dds"

	group = je_group_global_international_situations

	should_be_involved = {
		NOT = {
			is_involved_in_journal_entry = je_berlin_conference
		}
		OR = {
			hidden_trigger = {
				OR = {
					this ?= c:FRA
					this ?= c:GBR
					this ?= c:GER
					this ?= c:ITA
					this ?= c:RUS
					this ?= c:AUS
					this ?= c:SCA
					this ?= c:SWE
					this ?= c:UNL
					this ?= c:BEL
					this ?= c:IBE
					this ?= c:SPA
					this ?= c:POR
				}
			}
			AND = {
				is_in_geographic_region = geographic_region_europe
				country_rank = rank_value:great_power
			}
		}
	}

	modifiers_while_active = {
	}

	should_show_when_not_involved = {
		OR = {
			is_in_geographic_region = geographic_region_europe
			is_in_geographic_region = geographic_region_subsaharan_africa
			country_has_interest_marker_in_africa = yes
            AND = {
				top_overlord ?= {
					capital = {
						is_in_geographic_region = geographic_region_europe
					}
				}
				has_variable = newly_formed_colonial_nation_var
			}
		}
	}

	is_shown_when_inactive = {
		NOT = {
			has_global_variable = berlin_conference_ended
		}
		any_country_in_europe = {
            has_technology_researched = colonization
        }
	}

	possible = {
		any_country_in_europe = {
			has_technology_researched = civilizing_mission
			OR = {
				has_law = law_type:law_colonial_resettlement
				has_law = law_type:law_colonial_exploitation
			}
			country_rank >= rank_value:major_power
        }
		hidden_trigger = { has_global_variable = berlin_conference_happening }
	}

	immediate = {
		scope:journal_entry = {
			set_variable = remove_journal_icon
		}
		add_to_global_variable_list = { name = part_of_berlin_conf_list target = this }
		set_variable = visible_berlin_conf_treaty_article
		set_global_variable = { name = berlin_months_elapsed value = 0 }
	}

	immediate_all_involved = {
		assign_conference_gp_number_effect = yes
		add_to_global_variable_list = { name = part_of_berlin_conf_list target = root }
		set_variable = visible_berlin_conf_treaty_article
		if = {
			limit = { var:conference_gp_number <= 3 }
			add_to_global_variable_list = { name = berlin_conf_top3_gp_list target = root }
		}
		set_variable = { name = berlin_claimed_count value = 0 }
		if = {
			limit = { is_ai = yes  c:BEL ?= this  var:conference_gp_number >= 4 }
			set_variable = snatching_congo
		}
		if = {
			limit = { is_ai = yes  c:GER ?= this  var:conference_gp_number >= 4 }
			set_variable = snatching_east_africa
		}
		if = {
			limit = { is_ai = yes }
			trigger_event = { id = berlin_conference.49 }
		}
		if = {
			limit = { is_ai = yes  var:conference_gp_number = 1 }
			get_ahead_africa_conference = yes
		}
	}


	on_complete = {
		trigger_event = { id = berlin_conference.3 popup = yes }
		every_country = {
			limit = {
				is_involved_in_journal_entry = je_berlin_conference
			}
			hidden_effect = {
				trigger_event = { id = berlin_conference.3 popup = yes }
				random_scope_state = {
					limit = {
						state_region = s:STATE_LIBERIA
					}
					set_state_owner = c:LIB
				}
			}
		}
	}

	complete = {
		OR = {
		any_country = {
			custom_tooltip = {
				text = 7accept_berlin_conf_tt
				has_variable = accepted_conf_var
			}
			count >= 7
		}
		any_country = {
			custom_tooltip = {
				text = 3GPaccept_berlin_conf_tt
				has_variable = accepted_conf_var
				OR = {
					var:conference_gp_number = 3
					var:conference_gp_number = 2
					var:conference_gp_number = 1
				}
			}
			count >= 3
			} 
		}
	}

	fail = {
	}

	on_fail = {
	}

	event_outcome_failed_desc = {
	}

	on_timeout = {
	}

	event_outcome_timeout_desc = {
	}

	on_weekly_pulse = {
	}

	on_monthly_pulse = {
		events = {
			berlin_conference.50
			berlin_conference.51
		}
		effect = {
			set_global_variable = { name = berlin_claims_this_month value = 0 }
				random_country = {
					limit = {
						is_involved_in_journal_entry = je_berlin_conference
						is_ai = yes
					}
					trigger_event = { id = berlin_conference.52 }
					}
			
			clear_global_variable_list = part_of_berlin_conf_list
			every_country = {
				limit = { is_involved_in_journal_entry = je_berlin_conference }
				add_to_global_variable_list = { name = part_of_berlin_conf_list target = this }
			}
			clear_global_variable_list = berlin_conf_top3_gp_list
			every_country = {
				limit = {
					is_involved_in_journal_entry = je_berlin_conference
					var:conference_gp_number <= 3
				}
				add_to_global_variable_list = { name = berlin_conf_top3_gp_list target = this }
			}
		}
	}

	invalid = {
	}


	weight = 10000
	should_be_pinned_by_default = yes

	transferable = no
	can_revolution_inherit = no

}
