# French military JE: hard way = 100 battalions with ≥40% artillery; easy way = button with 30 battalions + all barracks fully employed, no shortages
ITA_French_militaryJE = {
	icon = "gfx/interface/icons/invention_icons/shell_gun.dds"
	group = je_group_italy_tyrone

	scripted_button = ita_french_military_complete_button

	possible = {
	}

	complete = {
		OR = {
			country_army_unit_type_fraction = {
				target = unit_type:combat_unit_type_mobile_artillery
				value >= 0.4
			}
			country_army_unit_type_fraction = {
				target = unit_type:combat_unit_type_shrapnel_artillery
				value >= 0.4
			}
		}
		OR = {
			# Hard way: 100 troops, at least 40% artillery (cannon or mobile)
			AND = {
				army_size >= 100
				any_scope_building = {
					filter = {
						is_building_type = building_barrack
					}
					NOT = { building_has_goods_shortage = yes }
					occupancy >= 0.9
					percent = 1
				}
			}
			

			# Easy way: click button + 30 units + all barracks fully employed, no shortages
			AND = {
				army_size >= 30
				any_scope_building = {
					filter = {
						is_building_type = building_barrack
					}
					NOT = { building_has_goods_shortage = yes }
					occupancy >= 0.9
					percent = 1
				}
				custom_tooltip = {
					text = ita_french_military_barracks_ok_tt
					has_variable = ita_french_military_complete_clicked
				}
			}
		}
	}

	invalid = {
		exists ?= c:ITA
	}

	on_complete = {
		set_variable = ita_french_military_done
		if = {
			limit = { has_variable = ita_french_military_complete_clicked }
			ita_stance_france_20_effect = yes
			add_modifier = {
				name = french_military_complete_easy_reward
				days = long_modifier_time
			}
		}
		else = {
			ita_stance_france_5_effect = yes
			add_modifier = {
				name = french_military_complete_artillery_reward
				days = long_modifier_time				
			}
		}
	}


	should_be_pinned_by_default = yes
}

# French economy JE: Sardinia GDP must grow 5% every 6 months for 5 periods in a row (Gründerzeit-style).
# Progress 0–5; miss one deadline = -1 progress. Variables on owner: baseline + 6‑month counter.
ITA_French_economyJE = {
	icon = "gfx/interface/icons/event_icons/ita_dev_JEs.png"
	group = je_group_italy_tyrone

	scripted_progress_bar = ita_french_economy_progress_bar
	scripted_button = ita_french_economy_complete_button

	possible = {
	}

	immediate = {
		# Baseline = Sardinia’s GDP at start; counter = first month of 6‑month window
		set_variable = {
			name = ita_french_economy_sar_baseline
			value = ita_sar_gdp
		}
		set_variable = {
			name = ita_french_economy_sar_goal
			value = {
				value = var:ita_french_economy_sar_baseline
				multiply = 1.05
			}
		}
		set_variable = ita_french_economy_month_1
	}

	on_monthly_pulse = {
		random_events = {
			60 = 0
			15 = italyEventsTyrone.7
			30 = italyEventsTyrone.8
		}
		effect = {
			# Advance 6‑month counter (month_1 -> … -> month_6 -> check then back to month_1)
			if = {
				limit = { has_variable = ita_french_economy_month_1 }
				remove_variable = ita_french_economy_month_1
				set_variable = ita_french_economy_month_2
			}
			else_if = {
				limit = { has_variable = ita_french_economy_month_2 }
				remove_variable = ita_french_economy_month_2
				set_variable = ita_french_economy_month_3
			}
			else_if = {
				limit = { has_variable = ita_french_economy_month_3 }
				remove_variable = ita_french_economy_month_3
				set_variable = ita_french_economy_month_4
			}
			else_if = {
				limit = { has_variable = ita_french_economy_month_4 }
				remove_variable = ita_french_economy_month_4
				set_variable = ita_french_economy_month_5
			}
			else_if = {
				limit = { has_variable = ita_french_economy_month_5 }
				remove_variable = ita_french_economy_month_5
				set_variable = ita_french_economy_month_6
			}
			else_if = {
				limit = { has_variable = ita_french_economy_month_6 }
				remove_variable = ita_french_economy_month_6
				set_variable = ita_french_economy_month_1

				# Goal = baseline × 1.05
				set_variable = {
					name = ita_french_economy_sar_goal
					value = {
						value = var:ita_french_economy_sar_baseline
						multiply = 1.05
					}
				}
