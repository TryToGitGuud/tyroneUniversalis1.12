REPLACE_OR_CREATE:je_communism_1 = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
		NOT = { has_variable = communism_1_done }
		OR = {
			has_technology_researched = socialism
			has_technology_researched = anarchism
		}
	}

	scripted_progress_bar = communism_1_progress_bar

	possible = {
		OR = {
			any_political_movement = {
				OR = {
					is_political_movement_type = movement_socialist
					is_political_movement_type = movement_anarchist
					is_political_movement_type = movement_communist
					is_political_movement_type = movement_internationale
				}
			}
			ig:ig_trade_unions ?= {
				ig_approval < -5
			}
		}
		
	}

	immediate = {
		root = {
			save_scope_as = root_country
		}
		set_communism_vars = yes
		trigger_event = { id = communism.10 days = 1 popup = yes }
	}

	on_weekly_pulse = {
		effect = {
			calculate_communism_progress = yes
		}
	}

	on_monthly_pulse = {
		random_events = {
			200 = 0
			5 = communism.7
		}
		effect = {
			calculate_communism_progress = yes
		}
	}

	fail = {
		OR = {
			custom_tooltip = {
				text = communism_progress_bar_at_full_tt
				scope:journal_entry = {
					"scripted_bar_progress(communism_1_progress_bar)" >= 100
				}
			}
			has_law_or_variant = law_type:law_council_republic
		}
	}

	on_fail = {
		set_variable = communism_1_done
		trigger_event = { id = intl_communism.2 popup = yes }
			if = {
				limit = {
					NOT = { has_variable = syndicalist_general_strike_done }
				}
				if = {
					limit = {
						any_interest_group = {
							law_stance = {
								law = law_type:law_council_republic
								value > neutral
							}
						}
					}
					random_interest_group = {
						limit = {
							law_stance = {
								law = law_type:law_council_republic
								value > neutral
							}
						}
						set_variable = communist_ig_var
					}
				}
				else = {
					ig:ig_trade_unions ?= {
						set_variable = communist_ig_var
					}
				}
				
			}
		
		custom_tooltip = communism_je_1_fail_tt
	}

	timeout = 1095

	on_timeout = {
		trigger_event = { id = communism.11 popup = yes }
		custom_tooltip = communism_je_1_complete_tt
	}

	should_be_pinned_by_default = yes
}

	is_shown_when_inactive = {

	}

	should_be_involved = {
		has_global_variable = intl_internationale_claimed
		country_rank >= rank_value:major_power
	}

	immediate_all_involved = {
		intl_update_internationale_alignment_lists = yes
		if = {
			limit = {
				has_global_variable = intl_internationale_leader_country
				this = global_var:intl_internationale_leader_country
			}
			set_variable = intl_internationale_leader
		}
		if = {
			limit = {
				NOT = { has_technology_researched = socialism }
			}
			add_technology_researched = socialism
		}
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				trigger = { has_global_variable = intl_great_proletarian_war_active }
				desc = je_the_internationale_status_great_war
			}
			triggered_desc = {
				trigger = { has_global_variable = intl_internationale_anarchist }
				desc = je_the_internationale_status_anarchist
			}
			triggered_desc = {
				trigger = { has_global_variable = intl_internationale_mutualist }
				desc = je_the_internationale_status_mutualist
			}
			triggered_desc = {
				trigger = { has_global_variable = intl_internationale_syndicalist }
				desc = je_the_internationale_status_syndicalist
			}
			triggered_desc = {
				trigger = { has_global_variable = intl_internationale_blanquist }
				desc = je_the_internationale_status_blanquist
			}
			triggered_desc = {
				trigger = { has_global_variable = intl_internationale_marxist }
				desc = je_the_internationale_status_marxist
			}
			triggered_desc = {
				trigger = {
					any_country = {
						country_rank = rank_value:great_power
						NOT = { intl_has_current_internationale_line = yes }
					}
				}
				desc = je_the_internationale_status_encircled
			}
			triggered_desc = {
				desc = je_the_internationale_status_final
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			intl_update_internationale_alignment_lists = yes
			if = {
				limit = {
					has_variable = intl_internationale_leader
				}
				intl_try_start_great_proletarian_war = yes
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = je_the_internationale_all_gps_tt
			NOT = {
				any_country = {
					country_rank = rank_value:great_power
					NOT = { intl_has_current_internationale_line = yes }
				}
			}
		}
	}

	on_complete = {
		trigger_event = { id = intl_communism.25 popup = yes }
	}

	weight = 1000
}

je_blanquist_consolidation = {
	icon = "gfx/interface/icons/law_icons/blanquist/blanquist_technocracy.dds"

	group = je_group_internationalism

	scripted_button = je_blanquist_reform_bureaucracy
	scripted_button = je_blanquist_reform_atheism
	scripted_button = je_blanquist_reform_economy
	scripted_button = je_blanquist_reform_state_police

	should_be_pinned_by_default = yes

	immediate = {
		set_variable = { name = blanquist_consolidation_timer value = 0 }
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_blanquist_consolidation_status_desc
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			change_variable = { name = blanquist_consolidation_timer add = 1 }
		}
	}

	complete = {
		custom_tooltip = {
			text = blanquist_consolidation_three_laws_tt
			OR = {
				AND = {
					has_law = law_type:law_blanquist_bureaucracy
					has_law = law_type:law_blanquist_state_atheism
					has_law = law_type:law_blanquist_economy
				}
				AND = {
					has_law = law_type:law_blanquist_bureaucracy
					has_law = law_type:law_blanquist_state_atheism
					has_law = law_type:law_secret_police
				}
				AND = {
					has_law = law_type:law_blanquist_bureaucracy
					has_law = law_type:law_blanquist_economy
					has_law = law_type:law_secret_police
				}
				AND = {
					has_law = law_type:law_blanquist_state_atheism
					has_law = law_type:law_blanquist_economy
					has_law = law_type:law_secret_police
				}
			}
		}
	}

	on_complete = {
		set_variable = blanquist_consolidation_done
		try_claim_internationale = { IDEOLOGY = blanquist }
		add_loyalists = {
			value = 0.25
		}
		every_political_movement = {
			limit = {
				is_political_movement_type = movement_internationale
			}
			add_modifier = {
				name = modifier_blanquist_movement_hegemony
			}
		}
	}

	fail = {
		var:blanquist_consolidation_timer >= 60
	}

	on_fail = {
		set_variable = blanquist_consolidation_failed
		if = {
			limit = { enacting_any_law = yes }
			cancel_enactment = yes
		}
		activate_law = law_type:law_presidential_republic
		activate_law = law_type:law_census_voting
	}

	current_value = {
		value = root.var:blanquist_consolidation_timer
	}

	goal_add_value = {
		value = 60
	}

	progressbar = yes

	weight = 1000
}

# Phase 1: only the first eligible country that takes this becomes the global "leader".
# Nobody else can start it until the leader is set. The leader gets the setup event (characters + faction pick).
# Phase 2: after the leader confirms the event, other countries receive je_mp_faction_world (not the setup event).

je_mp_faction_leader = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	group = je_group_mp_faction

	weight = 100
	should_be_pinned_by_default = yes

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_mp_faction_leader_desc
			}
		}
	}

	is_shown_when_inactive = {
	}

	possible = {
		hidden_trigger = {
			NOT = {
				has_global_variable = mp_faction_leader_country
			}
		}
		country_rank >= rank_value:great_power
		has_law_or_variant = law_type:law_council_republic
	}

	immediate = {
		set_variable = internationale_leader
		set_global_variable = {
			name = mp_faction_leader_country
			value = root
		}
		trigger_event = {
			id = mp_faction_events.1
			popup = yes
		}
	}

	complete = {
		has_variable = mp_faction_leader_confirmed
	}

	on_complete = {
		set_global_variable = mp_faction_phase_2
		every_country = {
			limit = {
				NOT = {
					this = global_var:mp_faction_leader_country
				}
				country_rank >= rank_value:major_power
				NOT = {
					has_journal_entry = je_mp_faction_world
				}
			}
			add_journal_entry = {
				type = je_mp_faction_world
			}
		}
	}

	invalid = {
		AND = {
			has_global_variable = mp_faction_leader_country
			NOT = {
				this = global_var:mp_faction_leader_country
			}
		}
	}
}
