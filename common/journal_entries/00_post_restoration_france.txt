#ORLEANIST

je_orleanist_france_industry_revolution = {
	icon = "gfx/interface/icons/event_icons/orleanist_main_icon.dds"
	group = je_group_historical_content

    is_shown_when_inactive = {
        exists = c:FRA
        c:FRA = ROOT
        has_variable = orleanist_restoration_var
        has_journal_entry = je_cement_the_rightful_dynasty
    }

    possible = {
        exists = c:FRA
        c:FRA = ROOT
        government_legitimacy >= 50
        OR = {
            custom_tooltip = {
                text = progress_cement_half_tt
                var:french_dynastic_primary_progress_var >= 60
            }
            custom_tooltip = {
                text = has_cemented_dynasty_TT
                has_variable = one_true_french_dynasty
            }            
        }
    }

	immediate = {
		set_global_variable = orleanist_eco_ongoing_global_var
		if = {
			limit = {
				NOT = {
					exists = global_var:orleanist_eco_ongoing_global_var_baseline
				}
			}
			set_global_variable = { 
				name = orleanist_eco_ongoing_global_var_baseline
				value = 0
			}
			c:FRA = {
				change_global_variable = { 
					name = orleanist_eco_ongoing_global_var_baseline
					add = this.gdp
				}
			}
		}
		if = {
			limit = {
				NOT = {
					exists = global_var:orleanist_eco_ongoing_global_var_actual
				}
			}
			set_global_variable = {
				name = orleanist_eco_ongoing_global_var_actual
				value = 0
			}
			c:FRA = {
				change_global_variable = { 
					name = orleanist_eco_ongoing_global_var_actual
					add = this.gdp
				}
			}
		}
		# Initial  GDP goal
		set_global_variable = { 
			name = orleanist_eco_ongoing_global_var_goal
			value = {
				value = global_var:orleanist_eco_ongoing_global_var_baseline
				multiply = 1.08 # 
			}
		}
		c:FRA = {
			save_scope_as = france_scope
		}
		c:GBR = {
			save_scope_as = gbr_scope
		}
		g:coal = {
			save_scope_as = coal_scope
		}
		g:iron = {
			save_scope_as = iron_scope
		}
		g:steel = {
			save_scope_as = steel_scope
		}


		
		set_variable = orleanist_france_variable

		scope:journal_entry = {
			add_modifier = orleanist_industry_modifier
		}

		c:FRA = {
			set_variable = {
				name = french_industry_var_progressbar
				value = 0
			}
		}
		c:FRA = { 
			set_variable = {
				name = french_industry_var_ministry_budget
				value = 150000
			}
		}
	}
	
	current_value = {
		value = var:french_industry_var_progressbar
	}

	goal_add_value = {
		add = 500
	}

	complete = {	
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		trigger_event ={
			id = orleanist_france.1
		}
		set_variable = orleanist_victory	
	}
	
	timeout = 10950
	

	fail = {
		
	}

	on_yearly_pulse = {
        effect = {
			c:FRA = {
				change_variable = {
					name = french_industry_var_ministry_budget
					add = 50000
				}
			}

            if = {
                limit = {
                    orleanist_actual_gdp_script_value >= global_var:orleanist_eco_ongoing_global_var_goal
                }
                if = {
                    limit = {
                        c:FRA = {
                            var:french_industry_var_progressbar < 500
                        }
                    }
                    c:FRA = {
                        change_variable = {
                            name = french_industry_var_progressbar
                            add = 25
                        }
                    }
                }
            }
			if = {
				limit = {
					exists = global_var:orleanist_eco_ongoing_global_var_baseline
				}
				set_global_variable = { # Redefines the baseline based on actual GDP
					name = orleanist_eco_ongoing_global_var_baseline
					value = orleanist_actual_gdp_script_value
				}
			}
			if = {
				limit = {
					exists = global_var:orleanist_eco_ongoing_global_var_baseline
					exists = global_var:orleanist_eco_ongoing_global_var_goal
				}
				# French GDP aim (updated quarterly)
				set_global_variable = { # Redefines goal based on baseline GDP
					name = orleanist_eco_ongoing_global_var_goal
					value = {
						value = global_var:orleanist_eco_ongoing_global_var_baseline
						multiply = 1.08 
					}
				}
			}			
        }
    }

	invalid = {
	}

	scripted_button = promote_investment_button
	scripted_button = advocate_mining_button
	scripted_button = encourage_production_innovation_button
	scripted_button = promote_construction_button

	weight = 1000
	should_be_pinned_by_default = yes

}

	scripted_button = return_of_ancien_regime
	scripted_button = enact_royal_censorship
	scripted_button = catholic_church_revival
	scripted_button = restore_noble_priviledges


	is_shown_when_inactive = {
		exists = c:FRA
		c:FRA = ROOT
		has_variable = legitimist_restoration_var
		OR = {
			has_journal_entry = je_cement_the_rightful_dynasty
			has_variable = one_true_french_dynasty
		}
	}

	possible = {
		OR = {
			custom_tooltip = {
				text = progress_cement_half_tt
				var:french_dynastic_primary_progress_var >= 60
			}
			custom_tooltip = {
				text = has_cemented_dynasty_TT
				has_variable = one_true_french_dynasty
			}			
		}
	}

	immediate = {
		trigger_event = {
			id = legitimist_events.1
		}
		set_variable = {
			name = legitimist_consolidation_var
			value = 0
		}
	}

	on_weekly_pulse = {
		effect = {
			clamp_variable = {
				name = legitimist_consolidation_var
				max = 120
				min = 0
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = legitimist_consolidation_var
				add = 1
			}	
		}
	}
	
	complete = {	
		custom_tooltip = {
			text = completed_legitimist_agenda_tt
			has_modifier = restored_noble_priviledges
			has_modifier = catholic_church_revived
			has_modifier = ancien_regime_restored
			has_law = law_type:law_outlawed_dissent
		}
	}

	on_complete = {
		activate_law = law_type:law_ancien_regime
		trigger_event = {
			id = legitimist_events.5
		}
	}

	fail = {
		NOT = {
			has_law = law_type:law_monarchy
		}
	}
	
	invalid = {
		country_rank < rank_value:great_power
	}

	current_value = root.var:legitimist_consolidation_var

	progressbar = yes

	goal_add_value = 120

	weight = 1000
	should_be_pinned_by_default = yes
		
}
