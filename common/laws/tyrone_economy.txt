law_state_capitalism = {
	group = lawgroup_economic_system
	
	icon = "gfx/interface/icons/law_icons/state_capitalism.dds"
	
	progressiveness = 100
	
	disallowing_laws = {
		law_serfdom
		law_anarchy
		law_women_in_the_fields
	}
	
	unlocking_technologies = {
		mutual_funds
	}
	
	unlocking_laws = {
		law_autocracy
		law_neo_absolutism
		law_oligarchy
		law_organic_regulation
		law_single_party_state
		law_technocracy
	}	
	
	on_activate = {
	}
	
	modifier = {
		country_private_construction_allocation_mult = 0.5
		building_nationalization_investment_return_add = 1
		country_government_dividends_reinvestment_add = 0.75
		country_government_dividends_efficiency_add = 0.5
		building_company_government_dividends_add = 0.5
		building_nationalization_cost_mult = -0.5
		building_nationalization_radicals_mult = -0.75
		country_max_companies_add = 2
		country_free_charters_add = 3
	}
	
	ai_will_do = {
		OR = {
			has_law_or_variant = law_type:law_technocracy
			AND = {
				has_law_or_variant = law_type:law_council_republic
				ruler ?= {
					has_ideology = ideology:ideology_vanguardist
					has_ideology = ideology:ideology_communist
				}			
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		
		if = {
			limit = { 
				has_law_or_variant = law_type:law_council_republic
			}
			add = 250
		}
		
		if = {
			limit = { 
				has_law_or_variant = law_type:law_technocracy
			}
			add = 100
		}
	}
	
	ai_impose_chance = {
		value = 0
		
		if = {
			limit = {
				has_law = law_type:law_command_economy			
			}
			
			add = base_impose_law_weight
			multiply = 10.0
		}
			
		if = {
			limit = {
				scope:target_country = {
					c:BIC ?= this
					OR = {	
						has_journal_entry = je_british_dictate_plantations
						has_journal_entry = je_british_dictate_military
						has_journal_entry = je_british_dictate_universities
					}
				}
			}
			multiply = 0
		}
	}	
}
