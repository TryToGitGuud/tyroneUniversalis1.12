# Sub-Saharan
berlin_conf_claim_west_africa = {
	scope = country
	is_shown = { is_involved_in_journal_entry = je_berlin_conference NOT = { has_global_variable = berlin_conference_ended } }
	is_valid = {
		custom_tooltip = {
			text = berlin_conf_is_involved_tt
			is_involved_in_journal_entry = je_berlin_conference
		}
		custom_tooltip = {
			text = berlin_conf_not_ended_tt
			NOT = { has_global_variable = berlin_conference_ended }
		}
		custom_tooltip = {
			text = berlin_conf_not_approving_tt
			NOT = { has_variable = accepted_conf_var }
		}
		hidden_trigger = { can_claim_west_africa = yes }
		custom_tooltip = {
			text = berlin_conf_area_locked_tt
			OR = {
				NOT = { has_global_variable = locked_west_africa }
				var:conference_gp_number <= 3
			}
		}
		custom_tooltip = {
			text = berlin_conf_has_claims_tt
			OR = {
				has_variable = claimed_west_africa_var
				var:remaining_claims_var >= 1
			}
		}
		custom_tooltip = {
			text = berlin_conf_max_claims_tt
			OR = {
				has_variable = claimed_west_africa_var
				NOT = { global_variable_list_size = { name = west_africa_claimed_countries_list value >= 3 } }
			}
		}
	}
	ai_is_valid = { always = no }
	effect = {
		custom_tooltip = berlin_claim_west_africa_states_tt
		if = {
			limit = { has_variable = claimed_west_africa_var }
			remove_variable = claimed_west_africa_var
			remove_list_global_variable = { name = west_africa_claimed_countries_list target = this }
			change_variable = { name = remaining_claims_var add = 1 }
		}
		else = {
			set_variable = claimed_west_africa_var
			add_to_global_variable_list = { name = west_africa_claimed_countries_list target = this }
			change_variable = { name = remaining_claims_var subtract = 1 }
		}
	}
}

		}

			change_variable = { name = remaining_claims_var add = 1 }
		}
	}
