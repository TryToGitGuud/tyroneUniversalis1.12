intl_cast_vote_current_sgui = {
	scope = country

	is_shown = {
		has_global_variable = intl_vote_active
	}

	is_valid = {
		NOT = { var:intl_vote_stance = 1 }
	}

	effect = {
		intl_cast_vote = { SIDE = 1 }
	}
}

	}

	is_valid = {
		NOT = { var:intl_vote_stance = 2 }
	}

	effect = {
		intl_cast_vote = { SIDE = 2 }
	}
}
