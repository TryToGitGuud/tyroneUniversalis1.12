on_game_start = {
	on_actions = {
		intl_parliament_init_on_action
	}
}

on_monthly_pulse_country = {
	on_actions = {
		intl_parliament_update_on_action
		intl_vote_monthly_pulse_on_action
	}
}

intl_parliament_init_on_action = {
	effect = {
		if = {
			limit = {
				is_player = yes
			}
			set_variable = { name = intl_strength_anarchist value = 20 }
			set_variable = { name = intl_strength_mutualist value = 20 }
			set_variable = { name = intl_strength_syndicalist value = 20 }
			set_variable = { name = intl_strength_blanquist value = 20 }
			set_variable = { name = intl_strength_marxist value = 20 }
			intl_update_parliament = yes
		}
	}
}

intl_parliament_update_on_action = {
	effect = {
		if = {
			limit = {
				is_player = yes
				has_journal_entry = je_internationale
			}
			intl_save_delegate_list = yes
			intl_save_delegate_chars = yes
			intl_update_parliament = yes
		}
	}
}

intl_vote_monthly_pulse_on_action = {
	effect = {
		if = {
			limit = {
				is_player = yes
				has_journal_entry = je_internationale
				intl_vote_is_active = yes
			}
			intl_update_vote_prestige = yes
			change_global_variable = { name = intl_vote_months_left add = -1 }
			if = {
				limit = { global_var:intl_vote_months_left <= 0 }
				intl_resolve_vote = yes
			}
		}
	}
}
